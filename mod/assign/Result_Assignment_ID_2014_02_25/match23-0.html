import java.util.Scanner;
import java.util.Arrays;

class Lab {

        String[][] data = new String[10][2];
        Scanner tallal = new Scanner(System.in);

        for (int i = 0; i < data.length; i++) {
            for (int j = 0; j < data[i].length; j++) {
