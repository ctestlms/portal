import java.awt.*;
import java.applet.*;

public class Pacman extends Applet
{
int xpos=10; int ypos=10;
int xtop=20; int ytop=20;
int xbot=370; int ybot=270;
int xinc=0; int yinc=0;

public void paint(Graphics g){
	g.setColor(Color.blue);
	if( (ypos<ybot) && (xpos<=xtop) ){
		g.fillArc( xpos, ypos, 25, 25, -45, 290);
		yinc=1;
		xinc=0;
		
	}
	if(ypos==ybot){
		g.fillArc( xpos, ypos, 25, 25, 45, 290);
		yinc=0;
		xinc=1;

	}
	
	if(xpos>=xbot){
		g.fillArc( xpos, ypos, 25, 25, 115, 290);
		yinc=-1;
		xinc=0;
		
	}
	if(ypos<=ytop)
		g.fillArc( xpos, ypos, 25, 25, 195, 290 );
	if((ypos<=ytop) && (xpos>=xbot) ){
		xinc=-1;
		yinc=0;
	}
	
	xpos+=xinc; ypos+=yinc;
	
	try{
			Thread.sleep(5);
			}
		catch(Exception e){}
	repaint(5);
	
}
}
