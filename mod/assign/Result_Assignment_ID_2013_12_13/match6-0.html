#include <iostream>
using namespace std;

class Node;
{
public:
	int a;
	Node* ptr;
	Node(int m, Node *x)
	{
		a=m;
		ptr=x;
	}
}



class linklist
{
private:
	Node *head;

public:
	linkedlist()
	{
		head=Null;
	}
	
	void addnode(int value)
	{
		Node *n;
		if(head==Null)
		{
			head=new Node(val, Null);
			n=head;
		}
		while(n->next!=Null);
		{
			n->next=new Node(val, Null);
		}
	}
}

int main()
{
	linklist list1;
	list1.addnode(1);
	list1.printnode();
}

void printnode()
{
	Node *n;
	n=head;
	while(n=!Null)
		cout<<"Data= "<<n->data<<endl;
	n=ptr;
}

void delnode(int index)
{
	Node *n;
	Node *delnode;
	Node *prevnode;
	Node *nextnode;
	n=head;
	for(int i=1; i<index; i++)
	{
		if(i==index-1)
		{
			prevnode=n;
			n=n->next;
		}
		delnode=n;
		nextnode=n->next;
		prevnode->next=nextnode;
		delete delnode;
	}
}
