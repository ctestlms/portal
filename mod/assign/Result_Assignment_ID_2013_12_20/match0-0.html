#include <iostream>
using namespace std;

class node
{
public:
	int dat;
	node* nextpointer;
	node(int d, node* nptr)
	{
		dat=d;
		nextpointer=nptr;
	}
};

class LINKEDLIST
{
private:
	node* headptr;
public:
	void newnode(int);
	void insertnode(int,int);
	void showlist();
	
	LINKEDLIST()
	{
		headptr=NULL;
	}

};
void LINKEDLIST::newnode(int val)
{
	node* n;
	n=headptr;
	if(headptr==NULL)
	{
		headptr=new node(val,NULL);
	}
	else
	{
		while(n->nextpointer!=NULL)
		{
			n=n->nextpointer;
		}
		n->nextpointer=new node(val,NULL);
	}
}
void LINKEDLIST::showlist()
{
	int i=0;
	node* n;
	n=headptr;
	while(n!=NULL)
	{
		cout<<"Node = "<<n->dat<<endl;
		n=n->nextpointer;
	}
}

void LINKEDLIST::insertnode(int val, int point)
{
	node* npointer;
	node* mynewnode;
	node* myprevnode;
	node* mynxtnode;
	npointer=headptr;
	for(int i=1;i<=point;i++)
	{
		if(i==point)
		{
			myprevnode=npointer;
		}
		npointer=npointer->nextpointer;
	}
	mynewnode=new node(val,npointer);
	mynxtnode=mynewnode->nextpointer;
	myprevnode->nextpointer=mynewnode;
}
