#include<stdio.h>

int main()
{
  int c,d,k,sum=0;
  int matrix1[3][3], matrix2[3][1], answer[3][1];
 
  printf("Enter the elements of matrix1\n");
 
  for(c=0;c<3 ;c++)
    for(d=0;d<3;d++)
      scanf("%d",&matrix1[c][d]);
  
    printf("Enter the elements of matrix2\n");
 
    for(c=0;c<3;c++)
      for(d=0;d<1;d++)
        scanf("%d",&matrix2[c][d]);
 
    for(c=0;c<3;c++)
    {
      for(d=0;d<1;d++)
      {
        for(k=0;k<3;k++)
        {
          sum=sum+matrix1[c][k]*matrix2[k][d];
        }
 
        answer[c][d]=sum;
        sum=0;
      }
    }
 
    printf("Product of entered matrices is\n");
 
    for(c=0;c<3;c++)
    {
      for(d=0;d<1;d++)
        printf("%d\t",answer[c][d]);
 
      printf("\n");
    }
getche();

}
