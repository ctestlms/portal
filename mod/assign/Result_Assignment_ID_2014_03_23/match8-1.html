#include<stdio.h>
#include<conio.h>
int main()
{
	float temp_f=0;

	float temp_c=0;
	float weight_p=0;
	float weight_k=0;
	float currency_pkr=0;
	float currency_usd=0;
	float area=0;
	char choice = 'X';
	

while (choice!='E')
{
	printf("\t\t CONVERTER APPLICATION\n");
	printf("A) Temperature (Fahrenheit - Celsius)\n");
	printf("B) Weight (Pound - Kilo)\n");
	printf("C) Currency (PKR - USD)\n");
	printf("D) Area (Mile - Kilometers)\n");
	printf("E) Terminate the application\n");
	scanf(" %c",&choice);

	switch(choice)
	{
	case'A':
		printf("Enter the temperature in Fahrenheit: ");
		scanf("%f",&temp_f);
		temp_c=(temp_f-32)/1.8;
		printf("Temperature in Celsius is: %1.2f\n", temp_c);
		break;
	case'B':
		printf("Enter the weight in Pounds: ");
		scanf("%f",&weight_p);
		weight_k=weight_p*0.453592;
		printf("Weight in Kilograms is: %1.3f\n", weight_k);
		break;
	case'C':
		printf("Enter the Currency in PKR: ");
		scanf("%f",&currency_pkr);
		currency_usd=currency_pkr/99;
		printf("Currency in USD is: %1.2f\n", currency_usd);
		break;
	case'D':
		printf("Enter the Area in Miles: ");
		scanf("%f",&area);
		area=area*2.59;
		printf("Area in Square Kilometers: %1.2f\n", area);
		break;
	case'E':
		printf("This application will terminate");
		break;
	default:
		printf("Wrong Input, Choose from the following\n Tip: All cases are in uppercase letters\n");
	}	
	
}
getche();
return 0;
}
