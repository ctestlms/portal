int bactFind(int a, int b, char** c, int rows, int columns)
{
	int count = 0;
	for (int i = -1; i <= 1; i++)					
	{
		if ((a + i) >= 0 && (a + i) < rows)
		{
			for (int j = -1; j <= 1; j++)
			{
				if ((b + j) >= 0 && (b + j) < columns)
				{
					if (i == 0 && j ==0 )		
						continue;
						if (c[a + i][b + j] == 'X')
							++count;
					
				}


			}
		}
	}
	return count;
}
