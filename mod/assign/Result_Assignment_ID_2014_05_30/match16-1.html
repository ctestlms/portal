#include<stdio.h>
#include<stdlib.h>
#include<conio.h>


int main()
{
	
	int *p;
	int	i=0;
	int n1=0;
	int n2=0;
	int avg=0;
	int sum=0;
	int count=0;
	char choice='x';
		
	while(choice!='q')
	{
		
			if(choice=='N')
			choice='q';
			else if (choice!='R')
			{
			printf("Enter A to allocate students marks or N to Exit\n");
			scanf(" %c",&choice);
			}
	
		switch (choice)
		{
		
			case 'q':
					avg=sum/count;
					printf("average=%d\n",avg);
					break;


			case 'A':
				printf("enter size of array: ");
				scanf("%d",&n1);

				p=(int*)malloc(n1*sizeof(int));
				
				for(i=0;i<n1;i++)
				{
					printf("Enter marks of student %d:\n",i+1);
				scanf("%d",p+i);
				count++;
				sum=sum+*(p+i);
				}
				printf("No more memory allocated. Do you want to reallocate?\nPress R to reallocate or N to Exit\n");
				scanf(" %c",&choice);
				break;

			case 'R':
					printf("enter new size of array:\n");
					scanf("%d",&n2);
					p=(int*)realloc(p,n2*sizeof(int));
					for(i=count;i<n2;++i)
					{	printf("Enter marks of student %d:\n",i+1);
						scanf("%d",p+i);
						sum=sum+*(p+i);
						count++;
					}
					printf("No more memory allocated. Do you want to reallocate?\nPress R to reallocate or N to Exit\n");
					scanf(" %c",&choice);
					break;
			case 'N':
				return 0;
			
			default:
				printf("Error! Re-input an option.\n");
	}
}
	getche();
	return 0;
}
