import java.awt.*;
import java.applet.*;
public class Pec extends Applet
	{
	int xbot = 370;
	int ybot = 270;
	int xpos=10; int ypos = 10;
	public void paint (Graphics g)
		{
		if (( ypos<= ybot) && (xpos==10)){		//while ball moves down
			ypos = ypos+1;
			g.fillArc(xpos,ypos,25,25,290,320);} 		//printing down pec

		else if(( ypos >= ybot) && (xpos <= xbot)){		//while ball moves towards right
			xpos = xpos+1;
			g.fillArc(xpos,ypos,25,25,20,320); }		//printing right pec

		else if(( xpos >= xbot) && (ypos>=10)){			//while ball moves up
			ypos = ypos-1;
			g.fillArc(xpos,ypos,25,25,110,320);}		//printing up pec

		else if ((ypos <=10)&&(xpos>=10)){		//while ball moves towards left
			xpos = xpos -1;
			g.fillArc(xpos,ypos,25,25,210,320);}		//printing left pec
		try{
			Thread.sleep(10);
			}
			catch(Exception e){}
		repaint(10); 
		}		//paint ends here
}		//class ends here
