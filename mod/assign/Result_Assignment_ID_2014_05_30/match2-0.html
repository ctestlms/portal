#define _CRT_SECURE_NO_WARNINGS
#include<stdio.h>
#include<conio.h>
#include<stdlib.h>
int main()										
{
	int *ptr, i = 0, j = 0, n1 = 0, n2 = 0, sum = 0;
	float avg = 0.0;
	char choice = 'r';

	printf("Enter the number of students in the class to calculate thier average:\n ");
	scanf("%d", &n1);
	ptr = (int*)malloc(n1*sizeof(int));

	for (i = 0; i<n1; i++)											
	{
		printf("\nto resume press 'r' and 'q' to quit right now\n");
		scanf("%c", &choice);
		if (choice != 'q')
		{
			printf("\nplease Enter marks of student%d: ", i + 1);
			scanf("%d", ptr + i);
			sum += *(ptr + i);
		}
		if (choice == 'q')
		{
			i++;
			break;
		}
	} 

	printf("\nYou have entered marks for the %d students\nPress 'r' to resume or 'q' to get average and quit", i);
	scanf("%c", &choice);

	if (choice != 'q')													
	{
		printf("\n\nEnter the size of array if you want to calculate the average of another class: ");
		scanf("%d", &n2);
		ptr = (int*)realloc(ptr, n2);
		printf("You can make %d more entries now\n", n2 - n1);
		for (j = 0; j < (n2 - n1); j++)
		{
			printf("Press 'r' to resume or 'q' to quit and get average\n");
			scanf("%c", &choice);
			if (choice != 'q')
			{
				printf("\nEnter marks of student %d: ", n1 + 1 + j);
				scanf("%d", ptr + i + j);
				sum += *(ptr + i + j);
			}
			if (choice == 'q')
			{
				j++;
				break;
			}

		}

	}

	avg = sum / (i + j);													
	printf("\nAverage=%1.1f", avg);
	getchar();
	return 0;
}
