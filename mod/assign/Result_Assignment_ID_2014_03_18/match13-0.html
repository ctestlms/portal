import java.awt.*;
import java.applet.*;

public class packman extends Applet{

int xpos=10; int ypos=50; 
int xinc=2; int yinc=2;
int xtop=20; int ytop=20; 
int xbot=370; int ybot=270; 
Color color;


public void init(){ 
color=Color.yellow;	
}
public Color genColor() {
return new Color((float)Math.random(),(float)Math.random(),(float)Math.random()); //for color changes randomly
}

	
public void paint(Graphics g)
{

if ((xpos>=xtop)){   
         color=genColor();
          xinc= -1;
 }
else if(xpos<=xbot) { 
             color=genColor(); 
             xinc=1; 
                }
if (ypos>=ybot) {
 color=genColor(); 
 yinc=-1;  
}
else if(ypos<=ytop) { 
color=genColor(); 
yinc=1; 
}
xpos+=xinc; ypos+=yinc;

g.setColor(color);
g.fillArc( xpos, ypos, 25, 25 , 310,280);
try{
			Thread.sleep(10);     //for slowdown
			}
		catch(Exception e){}
		repaint(10); //start again by removing previous one
 
}
}
