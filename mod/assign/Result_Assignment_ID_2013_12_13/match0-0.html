#include <iostream>

using namespace std;


class queues
{
 private:
    queues *head,*tail;
	int values;
	queues* lnk;

 public:
    queues()
    {
    	head=NULL;
    	tail=NULL;
    }

    int add(int x)
    {
    	queues *temp;
    	temp=new queues;
    	if(temp==NULL)
    	   cout<<"\nQUEUE FULL";

    	temp->values=x;
    	temp->lnk=NULL;
    	if(head==NULL)
    	{
    		tail=head=temp;
    		return x;
    	}
		else
           { tail->lnk=temp;
		tail=tail->lnk;}
		return x;
    }

    int del()
    {
    	if(head==NULL)
    	{
    		cout<<"\nQUEUE EMPTY";
    		return NULL;
    	}
        queues *temp;
        int n;
        n=head->values;
        temp=head;
        head=head->lnk;
        delete temp;
        return n;
    }

    ~queues()
    {
    	if(head==NULL)
    	   return;
    	queues *temp;
    	while(head!=NULL)
    	{
    		temp=head;
    		head=head->lnk;
    		delete temp;
    	}
     }
};

int main()
{
	queues Q;
	Q.add(11);
	Q.add(22);
	Q.add(33);
	Q.add(44);
	Q.add(55);
	cout<<"\nItem added= "<<Q.add(11);
	cout<<"\nItem added = "<<Q.add(12);
	cout<<"\nItem added = "<<Q.add(33);
	cout<<"\nItem added= "<<Q.add(44);
    cout<<"\nItem added= "<<Q.add(55);
	
	cout<<"\nItem Deleted = "<<Q.del();
	cout<<"\nItem Deleted = "<<Q.del();
	cout<<"\nItem Deleted = "<<Q.del();
	cout<<"\nItem Deleted = "<<Q.del();

	return 0;
}
