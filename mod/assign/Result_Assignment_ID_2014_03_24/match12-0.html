#include<stdio.h>
int main()
{

char choice = 'i';
	
while (choice!='E')
{
	printf("Enter a Character to enter converter (T for Temprature, W for Weight, C f0r Currecny and E to exit)\n");
	scanf(" %c",&choice);

	switch(choice)
	{
	case'T':
		float c;
		float f;
		printf("Enter temparature in celcius scale=");
		scanf("%f",&c);
		f=1.8*c+32;
		printf("Temparature in farnhite scale is:%fF\n",f);
		break;
	
	case'W':
		float p;
		float kg;
		printf("Enter the weight in pounds=");
		scanf("%f",&p);
		kg=0.453592*p;
		printf("Weight in kg is=%fkg\n",kg);	
		break;
	
	case'C':
		float pkr;
		float usd;
		printf("Enter the amount in PKR=");
		scanf("%f",&pkr);
		usd=pkr/99;
		printf("In USD the entered amount is=%f$\n",usd);
		break;



	default:
		printf("Wrong input =%c\n",choice);

	}
	
}
return 0;
}
