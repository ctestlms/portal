#include<stdio.h>
#include<conio.h>
void main()
{
	float F;
	float C;
	float P;
	float Kg;
	float PKR;
	float USD;
	float K;
	float miSq;
	float kmSq;
	float in_cube;
	float cm_cube;
	char Converter;
	Converter = 'X';
	printf("\nType\n t to start Conversion for Temp.\n C to start Conversion for Currency \n M to start Conversion for Mass\n A to start Conversion for Area\n V to start Conversion for Volume\n T to start conversion of Celsius to Kelvin\n e to exit converter\n");
	while(Converter!='e')
	{	
		scanf("%c",&Converter);
		switch(Converter)
		{	
			case 't':
				printf("\nENTER TEMPERATURE IN Celsius:");
				scanf("%f",&C);
				F = C*1.8;
				F=F+32;
				printf("\nTemperature in Fahrenheit is: %f", F);
				break;
			case 'M':
				printf("\nENTER VALUE IN Kgs:");
				scanf("%f",&Kg);
				P = Kg/0.453592;
				printf("\nMass in Pound is: %f",P);
				break;
			case 'C':
				printf("\nENTER VALUE IN PKR:");
				scanf("%f",&PKR);
				USD = PKR/99;
				printf("\nCurrnecy in USD is: %f",USD);
				break;
			case 'V':
				printf("\nEnter value in cubic centimeter:");
				scanf("%f",&cm_cube);
				in_cube= cm_cube/16.3871;
				printf("\nVolume in cubic inch is : %f", in_cube);
				break;
			case 'A':
				printf("\nEnter value in square kilometers:");
				scanf("%f",&kmSq);
				miSq= kmSq/2.58999;
				printf("\nArea in Square mile is: %f", miSq);
				break;
			case 'T':
				printf("\nEnter value in Celsius:");
				scanf("%f",&C);
				K= C + 273.16;
				printf("\nTemperature in Kelvin is: %f", K);
				break;
		}
	}
	getche();
}
