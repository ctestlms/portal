#include <stdio.h>
#include<conio.h>

int main()
{
  int m, n, p, q, c, d, k, sum = 0;
  int first[3][3], second[3][1], multiply[3][1];
 
  printf("Enter the elements of first matrix\n");
 
  for (  c = 0 ; c < 3 ; c++ )
    for ( d = 0 ; d < 3 ; d++ )
      scanf("%d", &first[c][d]);
 
 
	
    printf("Enter the elements of second matrix\n");
 
    for ( c = 0 ; c < 3 ; c++ )
      for ( d = 0 ; d < 1 ; d++ )
        scanf("%d", &second[c][d]);
 
    for ( c = 0 ; c < 3 ; c++ )
    {
      for ( d = 0 ; d < 1 ; d++ )
      {
        for ( k = 0 ; k < 3 ; k++ )
        {
          sum = sum + first[c][k]*second[k][d];
        }
 
        multiply[c][d] = sum;
        sum = 0;
      }
    }
 
    printf("Product of entered matrices:-\n");
 
    for ( c = 0 ; c < 3 ; c++ )
    {
      for ( d = 0 ; d < 1 ; d++ )
        printf("%d\t", multiply[c][d]);
 
      printf("\n");
    }
  
 
  return 0;
}
