#include<iostream>
using namespace std;

class node
{
public:
	int data;
	node* next;
	node(int a, node* b)
	{
		data=a;
		next=b;

	}
};

class linklist
{
private:
	node* head;
public:
	void addnode(int );
	void delnode(int );
	void insertnode(int,int );
	void printnode();
	linklist()
	{
		head=NULL;
	}
};


void linklist:: delnode(int index)
{
	node *n;
	node *delnode;
	node *prevnode;
	node *nextnode;
	n=head;
	for (int i=1; i<index; i++)
	{
		if (i ==index -1)
		{ prevnode=n;}
		n=n->next;
	}
	delnode=n;
	nextnode=delnode->next;
	prevnode->next=nextnode;
	delete delnode;
}

{
	
	node *n;

	node *nextnode;

	{
		if (i==index)

		n=n->next;
	}
	nextnode=n;
	insert=new node(val,NULL);
	prevnode->next=insert;

}
