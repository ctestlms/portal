#include<stdio.h>
#include<stdlib.h>
#include<conio.h>


int main()
{
	
	int *ptr;
	int	i=0;
	int n1=0;
	int n2=0;
	int avg=0;
	int sum=0;
	int c=0;
	char choice='x';
		
	while(choice!='q')
	{
		if(choice == 'Y')
			choice='r';
		else if(choice == 'E'||choice=='N')
		 {	
	     	choice='q';
		 }
			else
			{
			printf("Enter M to input Marks or E to Exit\n");
			scanf(" %c",&choice);
			}
		
		switch (choice)
		{

			case 'M':
				printf("Enter Array Size: ");
				scanf("%d",&n1);

				ptr=(int*)malloc(n1*sizeof(int));
				
				for(i=0;i<n1;i++)
				{
					printf("Input marks of student %d:\n",i+1);
				scanf("%d",ptr+i);
				sum=sum+*(ptr+i);
				c++;
				}
				printf("No more memory slots avaiable. Do you want to reallocate???\nPress Y/N. \n");
				scanf(" %c",&choice);
				break;

			case 'r':
					printf("New array size: ");
					scanf(" %d",&n2);
					ptr=(int*)realloc(ptr,n2*sizeof(int));
					for(i=c;i<n2;++i)
					{	printf("Input marks of student %d:\n",i+1);
						scanf("%d",ptr+i);
						sum=sum+*(ptr+i);
						c++;
					}
					printf("No more memory slots avaiable. Do you want to reallocate???\nPress Y/N.\n ");
					scanf(" %c",&choice);
					break;
			case 'q':
					avg=sum/c;
					printf("average=%d\n",avg);
					break;
	}
	}
	getche();
	return 0;
}
