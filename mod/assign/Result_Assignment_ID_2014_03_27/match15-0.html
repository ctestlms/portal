#include<stdio.h>
#include<conio.h>
int main()
{ 
	float Celsius,Farenhiet,pound,kilos,rupee,dollar;
	char choice='i';
	while (choice!='e')
	{
		printf("Enter a char t for tempertature,w for weight,c for currency, e to exit=");
		scanf("%c",&choice);

		switch(choice)
		{
					case 't':
						printf("Please enter temperature in Celsius: \n");
						scanf("%f",&Celsius);

						Farenhiet=(Celsius * 1.8) + 32;

						printf("Temperature in Farenhiet is %f \n", Farenhiet);
						break;
		
					case 'w':
						printf("Please enter weight in pounds: \n");
						scanf("%f",&pound);
						
						kilos=(pound*0.453);
						printf("weight in kilos is %f \n", kilos);
						break;
					case 'c':
						printf("Please enter currency in rupee: \n");
						scanf("%f",&rupee);
						dollar=(rupee/99);
						printf("currency in dollar is %f \n", dollar);
						break;

		}
	}
getchar();
return 0;
}
