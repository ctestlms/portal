#include <stdio.h>
#include <conio.h>

int main ()
{
	int a=0,b=0,n=0,d=0; //a=limit, b= number, n is counter for the for loop, d is for the final answer
	char r; //for choice
	printf ("Enter any character to proceed with the program. Enter 'q' for quitting the program\n");
	scanf ("%c", &r); 
	while (r != 'q') //provides the user with a chance to proceed or exit
	{//while bracket
		printf ("\nEnter a number\n");
		scanf ("%d", &b); //enter number
		printf ("\nEnter a limit\n");
		scanf ("%d", &a);	//enter limit for the table
		for (n=0; n<=a; n++) //counter should stop when it reaches the limit
		{//for bracket
			d=b*n;	
			printf("\n%d  *  %d  =  %d\n",b,n,d);
		}//for bracket
		printf ("\nEnter any character to proceed with the program. Enter 'q' for quitting the program\n");
		scanf ("%c", &r); //provides the user with a chance to proceed or exit

	}//while bracket
getchar ();
return 0;
}
