import java.awt.*;
import java.applet.*;

public class MoveBall extends Applet 
{ 
int xpos=10; int ypos=10; 
int xinc=0; int yinc=2;
int xtop=0; int ytop=20; 
int xbot=370; int ybot=270; 
Color color;


public void init()	{ 
color=Color.yellow;	
}
public Color genColor() {
return Color.yellow;
}

public void paint(Graphics g){
                    g.fillArc(xpos,ypos,20,20,-45,270);
         
		if (xpos==xtop){    yinc= 1; 
                  
                 if (ybot==ypos)
                                {
                                    xinc++;    
                                         }   
              }
		//if (ybot==ypos) {  xinc=1;  }
		
		xpos+=xinc; ypos+=yinc;
		g.setColor(color);
		
		try{
			Thread.sleep(20);
			}
		catch(Exception e){}
		repaint(10); 
}//paint ends 

}//class ends
