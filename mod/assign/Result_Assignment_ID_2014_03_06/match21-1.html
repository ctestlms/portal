///used to free char array
void freeArrayChar (char**arr,int row)
{

	for (int i=0;i<row;i++)
		free(arr[i]);
	free(arr);

	printf("free");
}
///used to free int array
void freeArrayInt (int**arr,int row)
{


	for (int i=0;i<row;i++)
		free(arr[i]);
	free(arr);
	printf("free");

}

//count the neighbour
void bacteriaCounter(char **arr,int **counter,int row,int col)

{
	for (int i=0;i<row;i++)
	{
		for (int j=0;j<col;j++)
		{
			counter[i][j]=0;
		}

	}
	for (int k=0;k<row;k++)
	{
		for (int l=0;l<col;l++)
		{
			for(int i=-1;i<2;i++)
			{

				if(k+i>=0 )
				{
					if(k+i<row)
					{

						for(int j=-1;j<2;j++)
						{	
							if(l+j>=0 )

							{
								if(l+j<col)
								{
									if(i==0&&j==0)
										continue;
									if(arr[k+i][l+j]=='X')
								
										counter[k][l]++;
									
								}
							}
						}

					}

				}


			}


		}

	}


}
//
void bacteriaChecker(char **arr,int **counter,int row ,int col)
{


	for (int i=0;i<row;i++)
	{
		for (int j=0;j<col;j++)
		{
			if (counter[i][j]<=1)
				arr[i][j]='-';
			if (counter[i][j]==3)
				arr[i][j]='X';
			if (counter[i][j]>=4)
				arr[i][j]='-';
		}
	}
}
///print the char array
void printFunc (char **arr,int row ,int col)
{


	for (int i=0;i<row;i++)
	{
		for (int j=0;j<col;j++)
		{
			printf("%c",arr[i][j]);

		}
		printf("\n");
	}



}

///create char dynamic array
char **CreateCharDynamicArray(int row,int col)

{
	char**arr=new char*[row];

	for (int i=0;i<row;i++)
		arr[i]=new char[col+1];

	return arr;

}
///create int dynamic array
int **CreateIntDynamicArray(int row,int col)

{
	int**arr=new int*[row];

	for (int i=0;i<row;i++)
		arr[i]=new int[col+1];

	return arr;

}
