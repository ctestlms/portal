#include<stdio.h>
#include<stdlib.h>
#include<conio.h>
int main()
{


	int *ptr, n1 = 0, i, m = 0;
	int sum = 0; int avg = 0;
	char choice = '0';
	while (choice!= 'q')
	{
		printf("Enter the number of students : ");
		scanf_s("%d", &n1);
		ptr = (int*)malloc(n1*sizeof(int));
		for (i = 0; i <= n1; i++)

		{
			if (i == n1)
			{
				printf("Do you want to add more number of students\n1- Yes\n2- No\n");
				scanf_s(" %d", &m);
				if (m == 1)
				{
					printf("Enter the new number of students : ");
					scanf_s(" %d", &n1);
					ptr = (int*)realloc(ptr, n1);
				}
				else continue;
			}
			printf("Enter the marks of student no. %d : ", i + 1);
			scanf_s("%d", &ptr[i]);
			sum = sum + ptr[i];
		}
		avg = sum / n1;
		printf("Average is %d : \n", avg);
		printf("Enter a char to continue or 'q' to quite\n");
		scanf_s(" %c", choice);
	}
	

	return 0;
}
