#include <stdio.h>

int main()
{
  int a, b, c, v, e, f, g, sum = 0;
  int first[3][3], second[3][1], multiply[3][1];
 
  printf("Enter elements of first matrix\n");
 
  for (  c = 0 ; c < 3 ; c++ )
    for ( v = 0 ; v < 3 ; v++ )
      scanf("%d", &first[c][v]);
 
 
	
    printf("Enter the elements of second matrix\n");
 
    for ( c = 0 ; c < 3 ; c++ )
      for ( v = 0 ; v < 1 ;v++ )
        scanf("%d", &second[c][v]);
 
    for ( c = 0 ; c < 3 ; c++ )
    {
      for ( v = 0 ; v < 1 ; v++ )
      {
        for ( e = 0 ; e < 3 ; e++ )
        {
          sum = sum + first[c][e]*second[e][v];
        }
 
        multiply[c][v] = sum;
        sum = 0;
      }
    }
 
    printf("Product of entered matrices:-\n");
 
    for ( c = 0 ; c < 3 ; c++ )
    {
      for ( v = 0 ; v < 1 ; v++ )
        printf("%d\t", multiply[c][v]);
 
      printf("\n");
    }
  
 
  return 0;
}
