void loadFile (FILE *fptr, char **A, int Num_Rows, int Num_Colmn)
{
	while (!feof(fptr)) {
		for (int i = 0; i < Num_Rows; i++)	{
			for (int j = 0; j < Num_Colmn; j++)	{
				fscanf(fptr, "%c" , &A[i][j]); 
			}
			fscanf(fptr,"\n");	
		}
	}
}
void shuffleArray (char **Array1 , char **Array2, int Num_Rows, int Num_Colmn)
{

	for (int i = 0; i < Num_Rows ; i++) {
		for (int j = 0; j < Num_Colmn; j++) {
			Array1[i][j] = Array2[i][j];
		}
	}
}
