#include <stdio.h>
#include <conio.h>
int main()
{

	int i, number, limit, product;
	char choice;
	printf ("Press any character to execute the program or press 'q' to quit the program: ");
	scanf (" %c", &choice);
	while (choice!='q')
	{
		printf ("Enter a Number: ");
		scanf ("%d", &number);
		printf ("Enter the Limit: ");
		scanf ("%d", &limit);
		for (i=0; i<=limit; i++)
		{
			product = number*i;
			printf ("%dx%d=%d\n",number,i,product);
		}
	printf ("\n\nPress any character to start or press 'q' to terminate the program");
	fflush(stdin);	
	scanf ("%c", &choice);
	}
	printf("\n\nThe program is exiting!");
	getche();
	return 0;
	
}
