#include <iostream>

using namespace std;




class queue
{
 private:
    queue *front,*rear;
	queue *link;
	int data;
 public:
    queue()
    {
    	front=NULL;
    	rear=NULL;
    }

    int add(int n)
    {
    	queue *tmp;
    	tmp=new queue;
    	if(tmp==NULL)
    	   cout<<"\nQUEUE FULL";

    	tmp->data=n;
    	tmp->link=NULL;
    	if(front==NULL)
    	{
    		rear=front=tmp;
    		return n;
    	}
		else
          {  rear->link=tmp;
		rear=rear->link;}
    return n;
	}

    int del()
    {
    	if(front==NULL)
    	{
    		cout<<"QUEUE EMPTY";
    		return NULL;
    	}
        queue *tmp;
        int n;
        n=front->data;
        tmp=front;
        front=front->link;
        delete tmp;
        return n;
    }

    ~queue()
    {
    	if(front==NULL)
    	   return;
    	queue *tmp;
    	while(front!=NULL)
    	{
    		tmp=front;
    		front=front->link;
    		delete tmp;
    	}
     }
};

int main()
{
	queue q;
	q.add(1);
	q.add(2);
	q.add(3);
	q.add(4);
	q.add(5);
	cout<<"implementing queues "<<endl;
	cout<<"\nItem added = "<<q.add(1);
	cout<<"\nItem added = "<<q.add(2);
	cout<<"\nItem added = "<<q.add(3);
	cout<<"\nItem added = "<<q.add(4);
	cout<<"\nItem added = "<<q.add(5)<<endl<<endl;
	cout<<"deletion will take place from the front as shown below "<<endl;
	cout<<"\nItem Deleted = "<<q.del();
	cout<<"\nItem Deleted = "<<q.del();
	cout<<"\nItem Deleted = "<<q.del();
	cout<<endl;
	system("pause");
	return 0;
}
