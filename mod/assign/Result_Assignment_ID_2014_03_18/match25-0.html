import java.awt.*;
import java.applet.*;

public class MoveBall extends Applet 
{ 
int xpos=10; int ypos=50; 
int xinc=2; int yinc=2;
int xtop=20; int ytop=20; 
int xbot=470; int ybot=470; 
Color color;



public void init()	{ 
color=Color.red;	
}
public Color genColor() {
return new Color((float)Math.random(),(float)Math.random(),(float)Math.random()); 
}
//public void change(){
//g.fillArc(xpos, ypos, 25, 25 , 30 , 300);
//}

public void paint(Graphics g){

         



                 if (xpos>=xbot)  
		 color=genColor();
  
		else if(xpos<=xtop) 
		color=genColor();

  
		if (ypos>=ybot) 
		 color=genColor();

   
		else if (ypos<=ytop) 
		 color=genColor();

  

if ( ypos<=ybot && xpos<=xtop)
{yinc = 1; 
ypos+=yinc;
g.fillArc(xpos, ypos, 25, 25 , 280  , 300);
}

else if  (xpos<=xbot && ypos>=ybot )
{xpos+=xinc;
g.fillArc(xpos, ypos, 25, 25 , 30 , 300);
}


else if ( xpos>=xbot && ypos>=ytop)
{xpos-=xinc;
g.fillArc(xpos, ypos, 25, 25 , 60 , 120);
}
else if (ypos<=ytop && xpos>=xtop )
{

ypos-=yinc;
g.fillArc(xpos, ypos, 25, 25 , 150 , 210);
}

		g.setColor(color);
			//g.fillArc(xpos, ypos, 25, 25 , 30 , 300);
		try{
			Thread.sleep(10);
			}
		catch(Exception e){}
		repaint(15); 
}//paint ends 

}//class ends
