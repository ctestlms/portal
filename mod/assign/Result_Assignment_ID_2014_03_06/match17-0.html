char** CmemAlo (int row, int col) 
{
	char** map = (char**) malloc (sizeof(char*)*row);
	for (int i = 0; i < row ; i++) {
		map[i] = (char*) malloc (sizeof(char)*col);
	}
	return map;
}
int** ImemAlo (int row, int col) 
{
	int** map = (int**) malloc (sizeof(int*)*row);
	for (int i = 0; i < row ; i++) {
		map[i] = (int*) malloc (sizeof(int)*col);
	}
	return map;
}
void mapping( FILE *ptr,char** source,int row, int col)
{
	char temp='l';
	for(int r=0;r<row;r++) {
		for(int c=0;c<col;c++) {
		source[r][c]=fgetc(ptr);
		}
		fscanf(ptr,"%c",&temp);
	}
	return;
}
int bactFind(int rowindex, int colindex, char** c, int rows, int columns) //Help taken from Danish 
{
    int count = 0;
    for (int i = -1; i <= 1; i++)
    {
        if ((rowindex + i) >= 0 && (rowindex + i) < rows)
        {
            for (int j = -1; j <= 1; j++)
            {
                if ((colindex + j) >= 0 && (colindex + j) < columns){
                    if (i == 0 && j == 0)
                        continue;
                    {
                        if (c[rowindex + i][colindex + j] == 'X')
                            ++count;
                    }
                }

            }
        }
    }
    return count;
}
void free_Character(char**map, int rows, int cols)
{
for (int i = 0; i < rows; i++) {
        delete (map[i]);
    }
    delete map;
}
void free_Integer(int**map, int rows, int cols)
{
for (int i = 0; i < rows; i++) {
        delete (map[i]);
    }
    delete map;
}
