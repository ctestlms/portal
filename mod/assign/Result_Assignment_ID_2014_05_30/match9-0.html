#include <stdio.h>
#include <stdlib.h>
#include <conio.h>
void main()
{
	int n1;
	int *p;
	int sum=0;
	int j=0;
	int average=0;
	char choice='a';
while (choice!='q')
	{
	printf("Enter the number of students\t");
	scanf("%d",&n1);
	p=(int*)malloc(n1*sizeof(int));
	
	for (int i=0;i<=n1;i++)
	{
		if (i==n1)
		{
		printf("Do you want to enter the marks of more students? Press\n1-Yes\n2-No\n");
		scanf("%d",&j);
		if(j==1)
		{
		printf("Enter new number of students\t");
		scanf("%d",&n1);
		p=(int*)realloc(p,n1);
		}
		else continue;
		}
		printf("Enter marks of Student %d\t",i+1);
		scanf("%d",&p[i]);
		sum=sum+p[i];	
	}
	average=sum/n1;
	
	printf("Please press\nr-Re-enter the marks?\nq-Quit the program and obtain the average\n");
	scanf(" %c",&choice);
}
	printf("Class Average: %d",average);
	getche();

}
