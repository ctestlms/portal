#include<stdio.h>
#include<stdlib.h>
#include<conio.h>

int main()
{
	
	int *pak;
	int	x1=0;
	int x2=0;
	int i=0;
	int total=0;
	int avg=0;
	int c=0;
	char choice='x';
		
	while(choice!='q')
	{
		if(choice=='Y')
			choice='r';
		else
		{	if(choice=='N')
			choice='q';
			else
			{
			printf("Press a to enter marks of students or N to Exit\n");
			scanf(" %c",&choice);
			}
		}
		switch (choice)
		{
			case 'a':
				printf("Enter the number of students: ");
				scanf("%d",&x1);

				pak=(int*)malloc(x1*sizeof(int));
				
				for(i=0;i<x1;i++)
				{
					printf("Enter marks of student %d:\n",i+1);
				scanf("%d",pak+i);
				total=total+*(pak+i);
				c++;
				}
				printf(" \nPress Y to enter marks of more students  or N to Exit\n");
				scanf(" %c",&choice);
				break;

			case 'r':
					printf("Enter the number of more students:\n");
					scanf("%d",&x2);
					pak=(int*)realloc(pak,x2);
					for(i=c;i<x2;++i)
					{	printf("Enter marks of student %d:\n",i+1);
						scanf("%d",pak+i);
						total=total+*(pak+i);
						c++;
					}
					choice='N';
			case 'q':
					avg=total/c;
					printf("Average=%d\n",avg);
					break;
	}
	}
	getche();
	return 0;
}
