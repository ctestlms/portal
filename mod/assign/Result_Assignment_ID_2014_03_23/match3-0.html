#define _CRT_SECURE_NO_WARNINGS

{

{

	scanf(" %c",&choice);


	switch(choice)
	{
	case'T': //for temperature//
		float c;    //input variable for ?C//

printf("Enter Temperature in celsius\n");   
scanf("%f",&c); //store the value given by user//
f=(c*1.8)+32; //formula to calcute ?F //

		break;
	case'W': //for Weight//

printf("Enter Weight in kilos\n");
scanf("%f",&k); //store value given by user//
p=0.453592*k; //formula to calculate pounds//
printf("Weight in Pounds is: %f\n", p);	
		break;
case'C': //for currency//
	float USD; //input variable for USD//

	break;
