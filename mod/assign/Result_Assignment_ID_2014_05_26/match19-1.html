	#include<stdio.h>
	#include<conio.h>

	void main()
{
	/*variables for counter in two matrices*/
		int i;
		int j;

	/*variables for arrays*/
		int a[3][3];
		int b[1][3];

	/*variables for output of resultant matrix*/
		int d;
		int e;
		int f;

	printf("Enter your first matrix\n\n");
		for (i=0;i<3;i++)
	{
		for (j=0;j<3;j++)
	{
		
		printf("Enter value in column%d row%d: ",i+1,j+1);
		scanf("%d",&a[j][i]);

	}
	}
		printf("\nEnter your second matrix\n");
		
		for (i=0;i<3;i++)
	{
		for (j=0;j<1;j++)
	{
		
		printf("Enter value in column%d row%d: ",i+1,j+1);
		scanf("%d",&b[j][i]);

	}
	}

		d=(a[0][0]*b[0][0])+(a[1][0]*b[0][1])+(a[2][0]*b[0][2]);
		e=(a[0][1]*b[0][0])+(a[1][1]*b[0][1])+(a[2][1]*b[0][2]);
		f=(a[0][2]*b[0][0])+(a[1][2]*b[0][1])+(a[2][2]*b[0][2]);
	
		printf("\nYour resultant matrix is\n");
		printf("{%d}\n",d);
		printf("{%d}\n",e);
		printf("{%d}\n",f);
		
		getch();

}
