#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<conio.h>

int main()
{
	int *ptr, n, n1=0, sum=0;
	char input[10];

	printf("Enter number of students whose marks you would like to enter : ");
	scanf("%d", &n);
	ptr=(int *)malloc(n * sizeof(int));

	printf("Enter the marks of student %d ('q' to quit) : ",n1+1);
	fflush(stdin);
	gets(input);
	while(strcmp(input, "q")!=0)
	{
		if(n1<n)
		{
			
			*(ptr+n1)=atoi(input);
			n1++;
		}
		else
		{
			n+=10;
			ptr=(int *)realloc(ptr, n);
		}
		printf("Enter the marks of student %d ('q' to quit) : ",n1+1);
		fflush(stdin);
		gets(input);
	}

	for(int i=0; i<n1; i++)
	{
		sum+=*(ptr+i);
	}

	float average=(float)sum/(n1);
	printf("Average is : %f", average);
	getch();
}
