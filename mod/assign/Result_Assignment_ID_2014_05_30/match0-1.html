#include<stdio.h>
#include<stdlib.h>

int main()
{
int size;
int i,j;
char press;
int *ptr;
int total=0;
int average;
int extra;
printf("KEYS\tFUNCTION\nq   \tterminate\na   \tenter the marks of students\nb   \textra students\n\n");
scanf("%c",&press);
while(press!='q')
switch(press)
{
case 'a':
	printf("enter the number of students of class\n");

	scanf("%d",&size);

ptr=(int*)malloc(size*sizeof(int));


for(i=0;i<size;i++)
{
	printf("enter the marks of student %d\n ",(i+1));

	scanf("%d",(ptr+i));

	total=*(ptr+i)+total;

}

    average=total/size;
	printf("%d",average);
	printf("if you want to enter more students' marks please enter b\n");
	break;
case 'b':
	printf("enter number of extra students\n");
	scanf("%d",&extra);
	ptr=(int*)realloc(ptr,extra);
	for(j=0;j<extra;j++)
	{
	printf("enter marks of students\n");
	scanf("%d",(ptr+j));
	total=*(ptr+j)+total;
	}
	size=size+extra;
	average=total/size;
	break;
default:
	printf("error\n");
	break;
}
if(press=='q')
	printf("average is %d\n",average);
return 0;

}
