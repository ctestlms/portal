#include <iostream>
using namespace std;


class node
{
public:
	int t;
	node* ntpointer;
	node(int d, node* nptr)
	{
		t=d;
		ntpointer=nptr;
	}
};

class LINKEDLIST
{
private:
	node* hptr;
public:
	void newnode(int);
	void insertnode(int,int);
	void showlist();
	
	LINKEDLIST()
	{
		hptr=NULL;
	}

};
void LINKEDLIST::newnode(int val)
{
	node* n;
	n=hptr;
	if(hptr==NULL)
	{
		hptr=new node(val,NULL);
	}
	else
	{
		while(n->ntpointer!=NULL)
		{
			n=n->ntpointer;
		}
		n->ntpointer=new node(val,NULL);
	}
}
void LINKEDLIST::showlist()
{
	int i=0;
	node* n;
	n=hptr;
	while(n!=NULL)
	{
		cout<<"Node = "<<n->t<<endl;
		n=n->ntpointer;
	}
}

void LINKEDLIST::insertnode(int val, int point)
{
	node* npt;
	node* addnode;
	node* pnode;
	node* furthernode;
	npt=hptr;
	for(int i=1;i<=point;i++)
	{
		if(i==point)
		{
			pnode=npt;
		}
		npt=npt->ntpointer;
	}
	addnode=new node(val,npt);
	furthernode=addnode->ntpointer;
	pnode->ntpointer=addnode;
}
int main()

}
