#include<iostream>

using namespace std;
class queue //class
{  private:
	int num;
    queue* next;
	queue*top;
	queue*bottom;

public:
	queue() //constructor
	{   
		top=NULL;
		bottom=NULL;
	}

	bool is_Empty() //if is empty
	{
		if (top==NULL)
		{
			return 1;
		}
		else
		{
			return 0;
		}
		
	}
	void addnode(int val) //insertion
	{
	    queue*p=new queue;
		p->num=val;
		p->next=NULL;

		if(is_Empty()) //In case of one node only

		{
			top=p;
			bottom=p;
		}
		else
		{
			bottom->next=p;
			bottom=p;
		}

	}

	void deletevalue() //delete node
	{
		if (is_Empty())
		{
			cout<<"Queue is empty"<<endl;
			
		}

		else if (top==bottom) 
		{
			top=bottom=NULL;
		}
		else
		{
			queue*p=top;
			top=top->next;
			delete p;
			p=NULL;
		}


	}

   void print()
	{
		queue*p;
		p=top;

		if(is_Empty()) 
		{
			cout<<"\nQueue is empty."<<endl;
		}
		while(p!=NULL)
		{
			cout<<p->num<<endl;
			p=p->next;
		}
	}
	~queue()
	{
		delete top;
		delete bottom;
		
	}


};
