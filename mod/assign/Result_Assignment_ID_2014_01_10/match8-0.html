#include <iostream>
using namespace std;

class Node
{
public :
	int data;
	Node *next;
	Node (int a, Node *b)
	{
		data = a;
		next = b;
	}
	Node()
	{
	}
};

class linkedlist
{
private:
	Node *head;
public:
	void addnode(int);
	void deletenode(int);
	void insertnode(int,int);
	void printnode();
	linkedlist()
	{
		head=NULL;
	}
};

void linkedlist::deletenode(int index)
{
	Node *n;
	Node *delnode;
	Node *prevnode;
	Node *nextnode;
	n=head;
	for (int i=1 ; i<index ; i++)
	{
		if (i == index-1)
		{
			prevnode=n;
		}
		n=n->next;
	}
	delnode=n ;
	nextnode=delnode->next;
	prevnode->next=nextnode ;
	delete delnode;
};

void linkedlist::printnode()
{
	Node *n;
	n=head;
	while (n!=NULL)
	{
		cout << "Data = " << n->data << endl;
		n=n->next;
	}
}

void linkedlist::addnode (int val)
{
	Node *n;
	n=head;
	if (head==NULL)
	{
		head = new Node(val,NULL);
	}
	else
	{
		while (n->next!=NULL)
		{
			n=n->next;
		}
		n->next=new Node(val,NULL);
	}
}

void linkedlist::insertnode(int val,int index)
{
	Node *n;
	Node *p;
	n=head;
	if (index==1)
	{
		head = new Node(val,head);
	}
	else
	{
		for (int a=0; a < index-1 ; a++)
		{
			p = n;
			n=n->next;
		}
		p->next = new Node(val,n);
	}
}
