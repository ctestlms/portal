import java.awt.*;
import java.applet.*;

public class Pacman extends Applet 
{ 
int xpos=0; int ypos=0; 
int xinc=2; int yinc=2; 
int xtop=20; int ytop=20; 
int xbot=370; int ybot=270; 
Color color;


public void init()	{ 
color=Color.red;	
}
public Color genColor() {
return new Color((float)Math.random(),(float)Math.random(),(float)Math.random()); 
}

public void paint(Graphics g){

         
		if (ypos<=ybot)  { 
                    color=genColor(); 
                    yinc= 1;
                    g.setColor(color); 
                    g.fillArc(xpos,ypos,25,25, 300, 300); 
                    ypos+=yinc;
                  }
                else if(ybot<=ypos) { 
                   color=genColor(); 
                   xinc=1; 
                   g.setColor(color); 
                   g.fillArc(xpos,ypos,25,25, 30, 300);
		   ypos=ypos;
                   xpos+=xinc;
                 }
                
		else if(ypos>=ytop & xpos>=xtop)  { 
                   color=genColor(); 
                   yinc= -1;
                   g.setColor(color); 
                   g.fillArc(xpos,ypos,25,25, 120, 300); 
                   xpos=xpos;
                   ypos+=yinc;
                 }
                    
          
    
 
		
                try{
			Thread.sleep(10);
			}
		catch(Exception e){}
		repaint(10); 
 }
}
