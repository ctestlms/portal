void gray_scale(void)
{
	cout<<"************************** Third Screen *****************************"<<endl<<endl
		<<"By NS M.Omer Arshad Reg#353 33-Mts-B"<<endl<<endl<<endl<<"Please Enter name of output gray image: (example amp.bmp) "<<endl;
		
	FILE* pFiler,*pFilew;
		pFiler=fopen(name,"rb");
		cin>>name2;
		pFilew=fopen(name2,"wb");
		if(pFilew==NULL){perror("Error opening file");}
		else{
				fread(&h1,14,1,pFiler);
				fread(&h2,40,1,pFiler);
				h2.biBitCount=8;
				fwrite(&h1,14,1,pFilew);
				fwrite(&h2,40,1,pFilew);

				{ int k=0;
				int palette[1024];
				int i,j;
				for(i=0;i<256;i++)
				{
				for(j=0;j<3;j++)
				{palette [k++]=i;
				}
				palette [k++]=0;
				}
				k=0;
				fseek(pFilew,1078,SEEK_SET);
				for(int count=0;count<1024;count++)
				fputc(palette[count],pFilew);
				}
				fseek(pFiler,54,SEEK_SET);
				while(!feof(pFiler))
			{	
				b=fgetc(pFiler);
				g=fgetc(pFiler);
				r=fgetc(pFiler);
				x=fgetc(pFiler);
				pixelvalue=(0.3*b)+(0.59*g)+(0.11*r);
				fputc(pixelvalue,pFilew);
			}
		fclose(pFiler);
		fclose(pFilew);
		}
		getch();
		for(int w=0;w<40;w++)
			cout<<'\n';
}
