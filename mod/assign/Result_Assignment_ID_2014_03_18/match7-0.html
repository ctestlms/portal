import java.awt.*;
import java.applet.*;

public class PacMan extends Applet
{
	int xpos=25, ypos=25, xtop=20, ytop=20, xbot=200, ybot=200, xinc=0, yinc=1;
	Color color;
	
	public void init(){
		color=Color.black;
	}


	public void paint(Graphics g){


		if(ypos>=ybot){
			yinc=0; xinc=1;
			g.fillArc(xpos,ypos,20,20,45,270);
		}

		else if(ypos<=ytop){
			yinc=0;  xinc=-1;
			g.fillArc(xpos,ypos,20,20,-135,270);
		}

		if(xpos>=xbot){
			xinc=0; yinc=-1;
			g.fillArc(xpos,ypos,20,20,135,270);
		}

		else if(xpos<=xtop){
			xinc=0;  yinc=1;
			g.fillArc(xpos,ypos,20,20,-45,270);
		}
		ypos+=yinc; 
		xpos+=xinc;
		
		
		g.setColor(color);
		g.fillArc(xpos,ypos,20,20,-45,270);
		try{
			Thread.sleep(10);
			}
		catch(Exception e){}
		
		repaint(10); 

	}

}
