import java.util.*;

public class Record {
	public static void main(String args[]) {
		
		String[][] data= new String[10][2]; 
		Scanner record = new Scanner(System.in);
		double total=0,ave, min=4, max=0;
		String check;
		double[] lessav= new double[10];
		Arrays.fill(lessav,0);
		int pos=1,flag=0; 
		for(int i=0; i<data.length; i++){
			System.out.println("\nEnter the name of student no. " + (i+1));
			data[i][0]=record.nextLine();
			for(int x=0; x<i; x++) {
				if (data[i][0].equalsIgnoreCase(data[x][0])) {
					System.out.println("Error! The given name already exist...");
					System.exit(1);
				}
			}
			System.out.println("Enter the CGPA of student no. " + (i+1));
			data[i][1]=record.nextLine();
			double g=Double.parseDouble(data[i][1]);
			if (!(g >= 0.00 && g <= 4.00)) {
				System.out.println("Error! Incorrect CGPA...");
				System.exit(1);
			}
			total+=g;
			lessav[i] = g;
		}
		
		ave = (double) total/data.length;
		
		System.out.println("Record of the students:-\n\n");
		System.out.println("Name\t\tCGPA");
		
		for(int i=0; i<data.length; i++){
			System.out.println(data[i][0]+"\t"+data[i][1]);
		}
		
		for(int i=0; i<data.length; i++){
            double y=Double.parseDouble(data[i][1]);
            if(y>max){
                max=y;
            }
            if(y<min){
                min=y;
            }
        }
		
		System.out.println("\nMinimum CGPA of students is "+min);
		System.out.println("Maximum CGPA of students is "+max);
		System.out.printf("Average of students CGPA is %.2f\n", ave);
		System.out.println("\nStudents having CGPA less than average are:");
		
		for(int i=0;i<data.length;i++){
			if (lessav[i] < ave){
				System.out.println(data[i][0]+"\t"+data[i][1]);
			}
		}
		
		System.out.println("\nEnter the name of student to check if its record is available or not:");
		check=record.nextLine();
		for(int i=0; i<data.length; i++){
			if(data[i][0].equalsIgnoreCase(check)){
				System.out.println("\nRecord found!\n");
				System.out.println(data[i][0] + "\t\t" + data[i][1] + "\n");
				
				double cgpa=Double.parseDouble(data[i][1]);
				for(int x=0; x<data.length; x++){
					double a=Double.parseDouble(data[x][1]);
					if(cgpa<a)
					pos++;
				}
				System.out.print("\nPosition of student "+data[i][0]+" in class is: " + pos + " / "+data.length);
				flag=1;
				break;
			}				
		}
		if(flag!=1){
			System.out.print("\nRecord not found!\n");
		}		
	}
}
