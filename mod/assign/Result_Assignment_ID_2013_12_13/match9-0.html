#include<iostream>
using namespace std;

struct member
{
	char alphabet ;
	member*node_pointer;
	
};

class node
{
	member*firstnode;
	member*lastnode;

public:
	node() 
	{
		firstnode=NULL;
		lastnode=NULL;
	}

	void insertion (char optout_alphabet) 
	{
		member*pointer=new member;
		pointer->alphabet=optout_alphabet;
		pointer->node_pointer=NULL;

		if(firstnode!=NULL) 
		{
			lastnode->node_pointer=pointer;
			lastnode=pointer;
		}
		else
		{
			
			firstnode=pointer;
			lastnode=pointer;
		}

	}

	void poping_out () 
	{
		if (firstnode!=NULL)
		{
			member*pointer=firstnode;
			firstnode=firstnode->node_pointer;
			delete pointer;
			pointer=NULL;
			
		}

		else if (firstnode==lastnode) 
		{
			firstnode=lastnode=NULL;
		}
		else
		{
			cout<<"\nQueue has no empty nodes."<<endl;
			
		}


	}

	void poping_out(char &optout_alphabet)
	{
		if (firstnode!=NULL)
		{
			optout_alphabet=firstnode->alphabet;
			poping_out();
		}
		else
		{
			
			cout<<"\nQueue is empty."<<endl;
		}
	}


	void display ()
	{
		member*pointer;
		pointer=firstnode;

		if(firstnode==NULL) 
		{
			cout<<"\nQueue list does not have any member."<<endl;
		}
		while(pointer!=NULL)
		{
			cout<<pointer->alphabet<<endl;
			pointer=pointer->node_pointer;
		}
	}


};

int main()
{
	node queue_alphabet;
	char  del_alphabet;
	cout<<"Queue list of characters by insertion is : "<<endl;
	queue_alphabet.insertion('F');
	queue_alphabet.insertion('I');
	queue_alphabet.insertion('Z');
	queue_alphabet.insertion('Z');
	queue_alphabet.insertion('A');
	queue_alphabet.display();
	cout<<"After taking out first member of list,queue list left behind is: "<<endl;
	queue_alphabet.poping_out(del_alphabet);
	queue_alphabet.display();
	cout<<"\Alphabet poped out from the list is :"<<del_alphabet<<endl;
	return 0;

	
}
