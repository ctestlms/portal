import java.awt.*;
import java.applet.*;

public class MoveBall extends Applet 
{ 
int xpos=10; int ypos=50; 
int xinc=0; int yinc=1;
int xtop=20; int ytop=20; 
int xbot=370; int ybot=270; 

Color color;


public void init()	{ 
color=Color.yellow;	
}
public Color genColor() {
return new Color((float)Math.random(),(float)Math.random(),(float)Math.random()); 
}

public void paint(Graphics g){

         
		if (ypos>=ybot)
		{ 
			ypos=ybot;
//			color=genColor(); 
			xinc=1;
//			yinc=0; 

		}
		
		if(xpos>=xbot ) 
		{ 
			xpos=xbot; 
//			xinc=0;
			yinc=-1; 
			
			
		}	

		if (ypos<=ytop) 
		{ 
			ypos=ytop;
			color=genColor();
//			yinc=0;  
			xinc=-1;
		}
		
		 if(xpos<=xtop && ypos<=ytop) 
		{ 
			xpos=xtop; 
			xinc=0;
			yinc=1; 
			
			ypos=40;
		
}		
		xpos+=xinc; 
		ypos+=yinc;
		g.setColor(Color.yellow);
		g.fillArc(xpos,ypos,30,30,45,270);
		try
		{
			Thread.sleep(10);
		}
		catch(Exception e){}
		repaint(10); 
}//paint ends 

}//class ends
