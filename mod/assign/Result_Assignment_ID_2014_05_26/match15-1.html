#include <stdio.h>
#include<conio.h>

int main()
{
  int a, b, k, sum = 0;
  int m[3][3], v[3][1], multiply[3][1];
 
  printf(" Enter the elements of first matrix\n");
 
  for (  a = 0 ; a < 3 ; a++ )
    for ( b = 0 ; b < 3 ; b++ )
      scanf("%d", &m[a][b]);
 
 
	
    printf("insert the elements of second matrix\n");
 
    for ( a = 0 ; a < 3 ; a++ )
      for ( b = 0 ; b < 1 ; b++ )
        scanf("%d", &v[a][b]);
 
    for ( a = 0 ; a < 3 ; a++ )
    {
      for ( b = 0 ; b < 1 ; b++ )
      {
        for ( k = 0 ; k < 3 ; k++ )
        {
          sum = sum +m[a][k]*v[k][b];
        }
 
        multiply[a][b] = sum;
        sum = 0;
      }
    }
 
    printf("Product of require entered matrices:-\n");
 
    for ( a = 0 ; a < 3 ; a++ )
    {
      for ( b = 0 ; b < 1 ; b++ )
        printf("%d\t", multiply[a][b]);
 
      printf("\n");
    }
  getche();
 
  return 0;
}
