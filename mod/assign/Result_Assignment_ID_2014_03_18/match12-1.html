import java.awt.*;
import java.applet.*;

public class PacMan extends Applet 
{
	int xpos=20; int ypos=20; 
	int xinc=2; int 
yinc=2;
	int xtop=20; int ytop=20; 
	int xbot=340; int ybot=240;

	public void paint(Graphics g){
		g.setColor(Color.yellow);
		if (ypos<ybot)
			{
			ypos+=yinc;
			g.fillArc(xpos, ypos, 50, 50, -60, 300);
			}
		if (ypos >= ybot & xpos < xbot)
			{
			xpos+=xinc;
			g.fillArc(xpos, ypos, 50, 50, 30, 300);
			}
		if (ypos >= ybot & xpos>= xbot){
			yinc = -2;
			ypos+=yinc;
			g.fillArc(xpos, ypos, 50, 50, 60, -300);
			}
		if (ypos < ybot & xpos > xbot){
			xinc = -2;
			xpos
+=xinc;
			g.fillArc(xpos, ypos, 50, 50, 30, 300);
			}
		try{
			Thread.sleep(10);
			}
		catch(Exception e){}
		repaint(10); 
	}
}
