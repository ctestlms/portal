#include <iostream>
using namespace std;

class Node
{
public:
	int data ;
	Node *next;
	Node ( int a , Node* temp )
	{
		data = a ;
		next = temp;
	}
};

class list
{
private:
	Node* start;
public:
	void addnode(int a)
	{
		Node* temp;
		temp = start;
		if(start == NULL)
		{
			start = new Node(a,NULL); 
		}
		else
		{
			while(temp->next!=NULL)
				temp = temp->next;
			temp->next = new Node(a,NULL);
		}
	}

	void insertnode(int a,int pos)
	{
		Node* temp;
		temp = start;
		Node* prev_node;
		Node* next_node;
		Node* new_node;
		for (int i=1; i<pos; i++)
			temp = temp->next;
		prev_node = temp;
		temp = temp->next;
		next_node = temp;
		new_node = new Node(a,NULL);
		prev_node->next = new_node;
		new_node->next = next_node;
	}

	void print_node()
	{
		Node *temp;
		temp = start;
		while(temp->next!=NULL)
		{
			cout << "Value: " << temp->data << endl;
			temp = temp->next;
		}
		cout << "Value: " << temp->data << endl;
	}

	list()
	{
		start = NULL;
	}

};

int main()
{
	int pos,data;
	list obj;
	obj.addnode(1);
	obj.addnode(2);
	obj.addnode(3);
	obj.addnode(4);
	obj.addnode(5);
	obj.print_node();
	cout << "Enter position for data: \n";
	cin >> pos;
	cout<< "Enter data: \n";
	cin >> data;
	obj.insertnode(data,pos);
	obj.print_node();
	return 0;
}
