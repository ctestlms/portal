#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<conio.h>


int main()
{
	int *arrayPointer, arraySize, index=0, sum=0;
	char input[10];

	printf("Enter number of students whose marks you would like to enter : ");
	scanf("%d", &arraySize);
	
	//Dynamic Memory Allocation
	arrayPointer=(int *)malloc(arraySize * sizeof(int));

	printf("Enter the marks of student %d ('q' to quit) : ",index+1);
	fflush(stdin);
	gets(input);
	while(strcmp(input, "q")!=0)
	{
		if(index<arraySize)
		{
			
			*(arrayPointer+index)=atoi(input);
			index++;
		}
		else
		{
			//Dynamic Memory Reallocation
			arraySize+=10;
			arrayPointer=(int *)realloc(arrayPointer, arraySize);
		}
		printf("Enter the marks of student %d ('q' to quit) : ",index+1);
		fflush(stdin);
		gets(input);
	}

	for(int i=0; i<index; i++)
	{
		sum+=*(arrayPointer+i);
	}

	float average=(float)sum/(index);
	printf("Average is : %f", average);
	getch();
}
