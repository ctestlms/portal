import java.util.*;
public class Students {

    public static void main(String[] args) {
        float max = 0;
        float min = 4;
        float avg = 0;
        float avrg = 0;
        float CGPA = 0;
        int rank = 1;
        float compare = 0;
        String name;
        String[][] data = new String[10][2];
        Scanner keybd = new Scanner(System.in);
        for (int i = 0; i < data.length; i++) {
            for (int j = 0; j < data[i].length; j++) {
                System.out.println("Enter the name of student No:" + (i + 1) + ":-");
                data[i][j] = keybd.next();
                for (int x = 0; x < i; x++) {
                    if (i == 0) {
                        break;
                    }
                    if (data[x][0].equals(data[i][0])) {
                        System.out.println("Entry already exists.");
                        return;
                    }
                }
                System.out.println("Enter the CGPA of student No:" + (i + 1) + ":-");
                data[i][j + 1] = keybd.next();
                float z = Float.parseFloat(data[i][j + 1]);
                if (z > 4.00) {
                    System.out.println("Wrong entry, CGPA is greater than 4:");
                    return;
                }

                j++;
            }
        }
        System.out.println("\nNAME\tCGPA");
        for (int k = 0; k < data.length; k++) {
            for (int l = 0; l < data[k].length; l++) {
                System.out.print("" + data[k][l] + " " + "\t");
            }
            System.out.print("" + "\n");
        }

        for (int a = 0; a < data.length; a++) {
            float c = Float.parseFloat(data[a][1]);
            if (c > max) {
                max = c;
            }
        }
        for (int b = 0; b < data.length; b++) {
            float d = Float.parseFloat(data[b][1]);
            if (d < min) {
                min = d;
            }
        }
        for (int e = 0; e < data.length; e++) {
            float f = Float.parseFloat(data[e][1]);
            avg += f;
            avrg = (avg / data.length);
        }
        System.out.println("\nMax CGPA=" + max +" "+ "Min CGPA=" + min + " " + "Average CGPA=" + avrg);
        System.out.println("\nThe students whose GPA is less than average are:");
        for (int g = 0; g < data.length; g++) {
            float h = Float.parseFloat(data[g][1]);
            if (h < avrg) {
                System.out.println("=>" + data[g][0]);
            }
        }
        System.out.println("\nEnter name of the student");
        name = keybd.next();
        for (int x = 0; x < data.length; x++) {
            if (data[x][0].equals(name)) {
                CGPA = Float.parseFloat(data[x][1]);
                for (int y = 0; y < data.length; y++) {
                    compare = Float.parseFloat(data[y][1]);
                    if (compare > CGPA) {
                        rank++;
                    }
                }
                System.out.println("\nThe CGPA of" + " " + (data[x][0]) + " " + "is" + " " + (data[x][1]) + " " + "and rank is" + " " + rank);
                return;
            }

        }
        System.out.println("Entry Doesn't exists");
    }
}
