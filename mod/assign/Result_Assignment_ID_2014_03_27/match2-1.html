#include<stdio.h>
#include<conio.h>

void main()
{

	float Fahr;
	float Celsius;
	float Pound;
	float Kilo;
	float PKR;
	float USD;
	float Kelvin;
	float Sq_Mile;
	float Sq_km;
	float cube_inch;
	float cube_cm;
	char Converter;
	Converter = 'X';
	printf("\nType\n T to initialize Temperature Converter\n C to initialize Currency Converter\n M to initialize Mass Converter\n A to initialize Area Converter\n V to initialize Volume Converter\n t to initialize speacial conversion of Celsius to Kelvin\n E to exit converter\n");
	while(Converter!='E')
	{	
		scanf("%c",&Converter);
		switch(Converter)
		{	
			case 'T':
				printf("\nENTER TEMPERATURE IN Celsius:");
				scanf("%f",&Celsius);
				Fahr = Celsius*1.8;
				Fahr=Fahr+32;
				printf("\nTemperature in Fahrenite is: %f", Fahr);
				break;
			case 'M':
				printf("\nENTER VALUE IN KILOs:");
				scanf("%f",&Kilo);
				Pound = Kilo/0.453592;
				printf("\nMass in Pound is: %f",Pound);
				break;
			case 'C':
				printf("\nENTER VALUE IN PKR:");
				scanf("%f",&PKR);
				USD = PKR/99;
				printf("\nCurrnecy in USD is: %f",USD);
				break;
			case 'V':
				printf("\nEnter value in cubic centimeter:");
				scanf("%f",&cube_cm);
				cube_inch= cube_cm/16.3871;
				printf("\nVolume in cubic inch is : %f", cube_inch);
				break;
			case 'A':
				printf("\nEnter value in square kilometers:");
				scanf("%f",&Sq_km);
				Sq_Mile= Sq_km/2.58999;
				printf("\nArea in Square mile is: %f", Sq_Mile);
				break;
			case 't':
				printf("\nEnter value of temperature in Celsius:");
				scanf("%f",&Celsius);
				Kelvin= Celsius + 273.16;
				printf("\nTemperature in Kelvin is : %f", Kelvin);
				break;
		}
	}
	getche();
}
