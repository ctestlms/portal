int main()									      
{
	int *ptr, i=0, j=0, n1=0, n2=0, sum=0;
	float avg=0.0;
	char choice='i';

	printf("Enter size of the array:\n ");
	scanf("%d",&n1);
	ptr=(int*)malloc(n1*sizeof(int));
	
	while (choice!='q' && i<n1)											//malloc	
	{
		printf("\nEnter marks of student%d: ",i+1);
		scanf("%d",ptr+i);
		sum+= *(ptr+i);
		i++;
		printf("\nPress 'i' to resume or 'q' to quit and get average\n");
		scanf("%c",&choice);
	} //end while
	
	printf("\nYou have entered marks for the %d students\nPress 'i' to make more enteries or 'q' to get average and quit: ",i);
	scanf("%c",&choice);

	if(choice!='q')												//realloc
	{
		printf ("\n\nEnter new size of the array: ");
		scanf("%d", &n2);
		ptr=(int*)realloc(ptr,n2);
		printf("You can now make %d more entries\n",n2-n1); 
		while(choice!='q' && j<(n2-n1))	
		{
		printf ("\nEnter marks of student %d: ",n1+1+j);
		scanf("%d", ptr+i+j);							//using address (ptr+i+j) so does not overwrite previous addresses 
		sum+= *(ptr+i+j);
		j++;
		printf("Press 'i' to resume or 'q' to quit and get average\n");
		scanf("%c",&choice);
		}//end while
	}//end if

	avg=sum/(i+j);													//average
	printf("\nAverage=%1.1f",avg);
	getchar();
	return 0;
}
