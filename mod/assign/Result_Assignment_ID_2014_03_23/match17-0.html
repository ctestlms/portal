#include<stdio.h>
#include<conio.h>
int main()
{
	
	char choice = 'X';


while (choice!='T')
{
	printf("\n------      CONVERTER APPLICATION    ------\n\n");
	printf("In upper case form,\n");
	printf("- enter 'T' for Temperature conversion: Fahrenheit to Celsius \n");		/*Main menu for the program*/
	printf("- enter 't' for Temperature conversion: Kelvin to Celsius\n");
	printf("- enter 'W' for Weight conversion: Pounds to Kilograms\n");
	printf("- enter 'C' for Currency conversion: PKR to USD\n");
	printf("- enter 'A' for Area conversion: Sqr Miles to Sqr Kilometers\n");
	printf("- enter 'V' for Volume conversion: Cubic Inches to Cubic Centimeters\n");
	printf("\nWant to exit?\nPress 'E' to Terminate the application!\n");
	scanf(" %c",&choice);

	switch(choice)
	{
	case'T':
		float temp_c; float temp_f;
		printf("Enter the temperature in Fahrenheit: ");		/*temp_c means Celcius, temp_f mean Farenheit*/
		scanf("%f",&temp_f);
		temp_c=(temp_f-32)/1.8;
		printf("Temperature in Celsius is: %1.4f\n\n", temp_c);
		break;
	case't':
		float temp_k; float temp_c2; 
		printf("Enter the temperature in Kelvin: ");		/*temp_k means Kelvin, temp_c2 means new value output of Celcius*/
		scanf("%f",&temp_k);
		temp_c2=temp_k-273.15;
		printf("Temperature in Celsius is: %1.4f\n\n", temp_c2);
		break;
	case'W':
		float lb; float kg;
		printf("Enter the weight in Pounds: ");			/*lb is the short form of pounds, kg is the short form of kilograms*/
		scanf("%f",&lb);
		kg=lb*0.453592;
		printf("Weight in Kilograms is: %1.4f\n\n", kg);
		break;
	case'C':
		float pkr; float dollars;
		printf("Enter the Currency in PKR: ");		/*pkr is Pakistani Rupees, dollars mean U.S dollars*/
		scanf("%f",&pkr);
		dollars=pkr/99;
		printf("Currency in USD is: %1.4f\n\n", dollars);
		break;
	case'A':
		float miles; float km;
		printf("Enter the Area in Square Miles: ");		/*miles mean square miles, km means square kilometres*/
		scanf("%f",&miles);
		km=miles*2.59;
		printf("Area in Square Kilometers: %1.4f\n\n", km);
		break;
	case'V':
		float inches; float cm; 
		printf("Enter the Volume in Cubic Inches: ");		/*inches mean cubic inches, cm means cubic centimetres*/
		scanf("%f",&inches);
		cm=inches*16.387064;
		printf("Area in Cubic Centimeters: %1.4f\n\n", cm);
		break;
	case'E':
		printf("\n\nThis application will now terminate! Press 'Enter' to continue...");		/*termination of the program. E for exit (terminate)*/
		break;
	default:
		printf("Wrong Input, Choose from the following\n Tip: All cases are in uppercase letters\n\n");	/*in case of wrong ouput*/
	}	
	
}
getche();
return 0;
}
