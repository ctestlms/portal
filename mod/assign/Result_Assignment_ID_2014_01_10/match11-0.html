#include <iostream>
using namespace std;

class node
{
public:
	int dat;
	node* nxtptr;
	node(int d, node* nptr)
	{
		dat=d;
		nxtptr=nptr;
	}
};

class linklist
{
private:
	node* head;
public:
	void addnode(int);
	void deletenode(int);
	void insertnode(int,int);
	void printnode();
	
	linklist()
	{
		head=NULL;
	}

};
void linklist::addnode(int val)
{
	node* n;
	n=head;
	if(head==NULL)
	{
		head=new node(val,NULL);
	}
	else
	{
		while(n->nxtptr!=NULL)
		{
			n=n->nxtptr;
		}
		n->nxtptr=new node(val,NULL);
	}
}
void linklist::printnode()
{
	int i=0;
	node* n;
	n=head;
	while(n!=NULL)
	{
		cout<<"Node = "<<n->dat<<endl;
		n=n->nxtptr;
	}
}
void linklist::deletenode(int dpos)
{
	node* n;
	node* delnode;
	node* prevnode;
	node* nxtnode;
	n=head;
	for(int i=1;i<=dpos;i++)
	{
		if(i==dpos)
		{
			prevnode=n;
		}
		n=n->nxtptr;
	}
	delnode=n;
	nxtnode=delnode->nxtptr;
	prevnode->nxtptr=nxtnode;
	delete delnode;
}
void linklist::insertnode(int insertval, int pos)
{
	node* n;
	node* addnode;
	node* prevnode;
	node* nxtnode;
	n=head;
	for(int i=1;i<=pos;i++)
	{
		if(i==pos)
		{
			prevnode=n;
		}
		n=n->nxtptr;
	}
	addnode=new node(insertval,n);
	nxtnode=addnode->nxtptr;
	prevnode->nxtptr=addnode;
}
int main()
{
	linklist mylist;
	for(int i=1;i<7;i++)
	{
		mylist.addnode(i);
	}
	cout<<"Original list is: "<<endl;
	mylist.printnode();
	int delp,insval,insp;
	cout<<"Please enter value for insertion along with its position respectively: "<<endl;
	cin>>insval>>insp;
	mylist.insertnode(insval,insp);
	cout<<"List after insertion: "<<endl;
	mylist.printnode();
	cout<<"Please enter position for deletion: "<<endl;
	cin>>delp;
	mylist.deletenode(delp);
	cout<<"List after deletion: "<<endl;
	mylist.printnode();
	return 0;
}
