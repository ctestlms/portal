int** Check(int row, int col, char ** Array)
{
	int counter=0;
	int** T_array;
	T_array=New_int(row,col);
	

	for (int i=0;i<row;i++)
	{
		for (int j=0;j<col;j++)
		{
			T_array[i][j] = 0;
			counter=0;
			for(int k= -1;k<= 1;k++)
			{
				for (int l=-1;l<=1;l++)

				{
					if (((i+k<0)||(i+k>=row))||((j+l<0)||(j+l>=col))||(k==0 && l==0))
						continue;
					if (Array[i+k][j+l]=='X')
						counter++;
					T_array[i][j]=counter;
				}
			}
		}
	}
	return T_array;
}
