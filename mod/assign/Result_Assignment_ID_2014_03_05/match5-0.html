void middlecheck(char **arr, int **arr1, int rows, int columns){
	int count = 0;
	for (int a = 1; a < (rows - 1); a++){
		for (int b = 1; b < (columns - 1); b++){
			if (arr[a + 1][b] == 'X'){
				count++;
			}
			if (arr[a - 1][b] == 'X'){
				count++;
			}
			if (arr[a][b + 1] == 'X'){
				count++;
			}
			if (arr[a][b - 1] == 'X'){
				count++;
			}
			if (arr[a + 1][b - 1] == 'X'){
				count++;
			}
			if (arr[a + 1][b + 1] == 'X'){
				count++;
			}
			if (arr[a - 1][b - 1] == 'X'){
				count++;
			}
			if (arr[a - 1][b + 1] == 'X'){
				count++;
			}
			arr1[a][b] = count;

			count = 0;
		}
	}
}

void cornercheck(char **arr, int **arr1, int row, int column){

	if (arr[0 + 1][0] == 'X'){

	}

	}

		count++;
	}

	if (arr[row - 1][column - 1] == 'X'){

	}

		count++;
	}
