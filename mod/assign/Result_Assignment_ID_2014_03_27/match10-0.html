#include <stdio.h>
#include<conio.h>


void main()
{
float temp;
float cel;
float pounds;
float kilos;
float pkr;
float usd;
char choice = 'i';
printf("Enter a Char (T for Temprature; W for Weight; C for Currecny = ");
while (choice!='e')
{
	scanf(" %c",&choice);


	switch(choice)
	{
		case 'T':
	// get input in Celsius and convert to F 		
			printf("Enter Temperature in Celsius: ");
			scanf("%f",&temp);
			cel = temp*9/5+32;
			printf("Temperature in Fahrenheit is = %f",cel); 
			break;
		case 'W':
		// get input in Pounds and convert to Kilo
			printf("Enter Weight in Pounds: ");
			scanf("%f",&pounds);
			kilos = 0.453592*pounds;
			printf("Weight in Kilos is = %f",kilos);
			break;
		case 'C':
		// get input in PKR and convert to USD
			printf("Enter Amount in PKR: ");
			scanf("%f",&pkr);
			usd = pkr/99; // Keeping 1 USD = 99 PKR
			printf("Amount in USD is = %f",usd);
			break;

		default:
			printf("Wrong input =%c\n",choice);
	}
	
}
  getche();

}
