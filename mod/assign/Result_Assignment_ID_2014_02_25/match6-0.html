import java.util.Scanner;
import java.util.Arrays;
public class Record
{
	public static void main (String args[])
	{
		Scanner entry = new Scanner (System.in);
		String [][]record = new String [5][2];
		float []CGPA = new float[10];
		int counter=0;
		float sum = 0, avg = 0, min = 0, max = 0;
		for (int i=0;i<record.length;i++)
		{
			System.out.printf ("Student No. %d's name: ",i+1);
			record[i][0] = entry.nextLine();
			if (i>0)
			{
				for (int k=0;k<=counter;k++)
				{
					if (i==k)
					{
						continue;
					}
					if (record[i][0].equalsIgnoreCase(record[k][0]))
					{
						
						System.out.println ("Duplicate Entry");
						return;
					}
				}
			}
			System.out.printf ("CGPA of Student No. %d: ",i+1);
			record[i][1] = entry.nextLine();
			CGPA[i] = Float.parseFloat(record[i][1]);
			if ((CGPA[i]>4)||(CGPA[i]<0))
			{
				System.out.printf ("Invalid CGPA\nProgram Terminated");
				return;
			}
			counter++;
		}
		counter = 0;
		min = CGPA[0];
		max = CGPA[0];
		for (int i=0;i<record.length;i++)
		{
			sum = sum + CGPA[i];
			counter++;
			if (CGPA[i]<=min)
			{
				min = CGPA[i];
			}
			if (CGPA[i]>=max)
			{
				max = CGPA[i];
			}
		}
		avg = (sum)/counter;
		System.out.printf ("\n%s%10s\n","Name","CGPA");
		for (int i=0;i<record.length;i++)
		{
			for (int j=0;j<record[i].length;j++)
			{
				System.out.printf ("%15s",record[i][j]);
				System.out.printf ("%10s",record[i][++j]);
				System.out.println ("");
			}
		}
		System.out.printf ("Minimum CGPA: %.2f\n",min);
		System.out.printf ("Maximum CGPA: %.2f\n",max);
		System.out.printf ("Average CGPA: %.2f\n",avg);


		for (int i=0;i<record.length;i++)
		{
			if (CGPA[i]<avg)
			{
				System.out.printf ("Student number %d below the average.\n",(i+1),record[i][0]);
			}
		}
	}
}
