import java.awt.*;
import java.applet.*;

public class Move extends Applet 
{ 
int xpos=0; int ypos=0; 
int xinc=2; int yinc=2;
int xtop=20; int ytop=20; 
int xbot=370; int ybot=270; 

public void paint(Graphics g){
                g.setColor(Color.red);
		

                if (ypos<=ybot)
		{ypos+=yinc;
                g.fillArc(xpos,ypos,25,25,-60,300);}
                
                 else  if ((ypos>=ybot)&(xpos<=xbot))
		   
                      {
                        g.fillArc(xpos,ypos,25,25,30,300);
                        xpos+=xinc;
		      }
                    else if ((ypos<=ybot)&(xpos==xbot)&(ypos>=0))
		   
                      { 
                        g.fillArc(xpos,ypos,25,25,60,-300);
                        ypos-=yinc;
		      
}
                  
                 
		  
                      
		try{
			Thread.sleep(10);
			}
		catch(Exception e){}
		repaint(10); 
}//paint ends 

}//class endsja
