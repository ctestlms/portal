void Corners(char **pp1,char **pp2,int a,int b)
{
    //1st corner
    int counter=0;
    if(pp1[0][1]=='X')
        ++counter;
    if(pp1[1][1]=='X')
        ++counter;
    if(pp1[1][0]=='X')
        ++counter;
    if(counter==1 || counter==0)
    {
        pp2[0][0]='-';
    }
    else
        if(counter==3){
            pp2[0][0]='X';
        }//2nd corner
        counter=0;
        if(pp1[a][1]=='X')
            ++counter;
        if(pp1[a-1][1]=='X')
            ++counter;
        if(pp1[a-1][0]=='X')
            ++counter;
        if(counter==1 || counter==0)
            pp2[a][0]='-';
        else
            if(counter==3)
                pp2[a][0]='X';
        //3rd corner
        counter=0;
        if(pp1[0][b-1]=='X')
            ++counter;
        if(pp1[1][b-1]=='X')
            ++counter;
        if(pp1[1][b]=='X')
            ++counter;
        if(counter==1 || counter==0)
            pp2[0][b]='-';
        else
            if(counter==3)
                pp2[0][b]='X';
        //4rth corner
        counter=0;
        if(pp1[a][b-1]=='X')
            ++counter;
        if(pp1[a-1][b-1]=='X')
            ++counter;
        if(pp1[a-1][b]=='X')
            ++counter;
        if(counter==1 || counter==0)
            pp2[a][b]='-';
        else
            if(counter==3)
                pp2[a][b]='X';
        return;
}


void edge(char **pp1,char **pp2,int a,int b)
{
    int counter=0;

    for(int i=1;i<b;++i)

    {
      
        if(pp1[0][i-1]=='X')
            counter++;
        if(pp1[0][i+1]=='X')
            counter++;
        if(pp1[1][i]=='X')
            counter++;
        if(pp1[1][i-1]=='X')
            counter++;
        if(pp1[1][i+1]=='X')
            counter++;
        if(counter==1 || counter==0 || counter ==4)
            pp2[0][i]='-';
        else
            if(counter==3)
                pp2[0][i]='X';

        counter=0;
        if(pp1[a][i-1]=='X')
            counter++;
        if(pp1[a][i+1]=='X')
            counter++;
        if(pp1[a-1][i]=='X')
            counter++;
        if(pp1[a-1][i-1]=='X')
            counter++;
        if(pp1[a-1][i+1]=='X')
            counter++;
        if(counter==1 || counter==0 || counter==4)
            pp2[a][i]='-';
        else
            if(counter==3)
                pp2[a][i]='X';
    }

    for(int i=1;i<a;++i)

    {

        counter=0;
        if(pp1[i-1][b]=='X')
            counter++;
        if(pp1[i+1][b]=='X')
            counter++;
        if(pp1[i-1][b-1]=='X')
            counter++;
        if(pp1[i][b-1]=='X')
            counter++;
        if(pp1[i+1][b-1]=='X')
            counter++;
        if(counter==1 || counter==0 || counter==4)
            pp2[i][b]='-';
        else
            if(counter==3)
                pp2[i][b]='X';

        counter=0;
        if(pp1[i-1][0]=='X')
            counter++;
        if(pp1[i-1][1]=='X')
            counter++;
        if(pp1[i][1]=='X')
            counter++;
        if(pp1[i+1][1]=='X')
            counter++;
        if(pp1[i+1][0]=='X')
            counter++;
        if(counter==1 || counter==0 || counter==4)
            pp2[i][0]='-';
        else
            if(counter==3)
                pp2[i][0]='X';
        return;
    }

}



void Center(char **pp1,char **pp2,int a,int b)
{
    for(int i=1;i<a;++i)
    {
        for(int y=1;y<b;++y)
        {
            int counter=0;
            if(pp1[i][y+1]=='X')
                counter++;
            if(pp1[i][y-1]=='X')
                counter++;
            if(pp1[i+1][y-1]=='X')
                counter++;
            if(pp1[i-1][y-1]=='X')
                counter++;
            if(pp1[i+1][y]=='X')
                counter++;
            if(pp1[i+1][y+1]=='X')
                counter++;
            if(pp1[i-1][y+1]=='X')
                counter++;
            if(pp1[i-1][y]=='X')
                counter++;
            if(counter==1 || counter==0 || counter==4)
                pp2[i][y]='-';
            else
                if(counter==3)
                    pp2[i][y]='X';        
        }
    }

    return;

} //End of Main Function
