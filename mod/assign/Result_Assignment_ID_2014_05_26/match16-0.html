{

	int i=0;
	int j=0;
	int m=0;
	
	for (i=0;i<3;i++)
	{
		printf("First Matrix Row=%d\n", (i+1));
		for (j=0;j<3;j++)
		{
			printf("Column \t%d\t=\t", (j+1));
			scanf("%d", &num[i][j]);
		}
	}
	printf("Second matrix Row 1\n");
	for (m=0;m<3;m++)
	{
		printf("Column \t%d\t=\t",(m+1));
		scanf("%d", &second[m][0]);
	}
	 
		answer[0][0]=num[0][0]*second[0][0]+num[0][1]*second[1][0]+num[0][2]*second[2][0];

		answer[2][0]=num[2][0]*second[0][0]+num[2][1]*second[1][0]+num[2][2]*second[2][0];
		printf("\t\t\tTHE RESULTANT MATRIX\n");
		printf("\t\t\t(%d)\n",answer[0][0]);

}
