import java.util.*;

public class sorttin{
	
	public static void main(String args[]){

		Scanner kb = new Scanner(System.in);
		String arr[][] = new String[10][2];
		float gpa[] = new float[10];
		float avg = 0, min = 4, max = 0, sum = 0;
		String temp;
		float n;
		int rank = 0;
		
		for(int i = 0; i< 10; i++)
		{
			System.out.print("Enter your name: ");		//Inputting Name
			arr[i][0] = kb.nextLine();
	
			for(int j = i -1 ; j >= 0 ; j--)			//comparing current name with other names
			{
				if(arr[i][0].equalsIgnoreCase(arr[j][0]))
				{
					System.out.println("Name already exists... Program is exiting");
					return ;
				}
			}
			System.out.print("Enter Your CGPA: ");			//Inputting CGPA
			arr[i][1] = kb.nextLine();
			gpa[i] = Float.parseFloat(arr[i][1]);

			if( !( gpa[i] >= 0 && gpa[i] <= 4)){
				System.out.println("CGPA is invalid... Program is exiting");
				return;
			}
			
			sum += gpa[i];
			if( gpa[i] < min)
				min = gpa[i];
			if( gpa[i] > max)
				max = gpa[i];
		}
		avg = sum /10;
		

		//Printing the data
			System.out.println("\n\nNames\t\tCGPA");
			
		for(int i = 0; i < 10 ; ++i)
		{
			System.out.println("" + arr[i][0] + "\t\t" + arr[i][1]);
		}
		System.out.println("\nAverage = " + avg);
		System.out.println("Maximum CGPA = " + max);
		System.out.println("Minimum CGPA = " + min + "\n");

//===================== Printing below avg students ===========================

		
		System.out.println("\n\nNames of Students below Average : \n");
			
		for(int i = 0; i < 10 ; ++i)
		{	
			n = Float.parseFloat(arr[i][1]);
			if(n < avg)
				System.out.println("Name : " + arr[i][0] + "\t CGPA: " + arr[i][1]);
		}
//====================== Sorting the array containing GPAs to find Rank====================

		Arrays.sort(gpa);
		
		System.out.println("\nEnter name of Student: ");
		temp = kb.nextLine();

		for(int i = 0; i < 10 ; i++)
		{
			if(temp.equalsIgnoreCase(arr[i][0]))
			{
				n = Float.parseFloat(arr[i][1]);
				for(int j = 9; j >= 0 ; j--)		//checking in reverse order because Highest rank appers at the end in sorted array
				{
					if( n == gpa[j])
					{	
						rank = 10 - j ;
						break;
					}
				}
				System.out.println("Name: " + arr[i][0] + " CGPA: " + arr[i][1] + " Rank in class: " + rank);
				break;
			}
			else
			{	
				System.out.println("This student data is not available");
			}
		}

	}

}
