void kona(char **gola,char **bhola,int a,int b)
{
	int count=0;
	if(gola[0][1]=='X')
		++count;
	if(gola[1][1]=='X')
		++count;
	if(gola[1][0]=='X')
		++count;
	if(count==1 || count==0)
	{
		bhola[0][0]='-';
	}
	else
		if(count==3){
			bhola[0][0]='X';
		}
		count=0;
		if(gola[a][1]=='X')
			++count;
		if(gola[a-1][1]=='X')
			++count;
		if(gola[a-1][0]=='X')
			++count;
		if(count==1 || count==0)
			bhola[a][0]='-';
		else
			if(count==3)
				bhola[a][0]='X';
		
		count=0;
		if(gola[0][b-1]=='X')
			++count;
		if(gola[1][b-1]=='X')
			++count;
		if(gola[1][b]=='X')
			++count;
		if(count==1 || count==0)
			bhola[0][b]='-';
		else
			if(count==3)
				bhola[0][b]='X';
		
		count=0;
		if(gola[a][b-1]=='X')
			++count;
		if(gola[a-1][b-1]=='X')
			++count;
		if(gola[a-1][b]=='X')
			++count;
		if(count==1 || count==0)
			bhola[a][b]='-';
		else
			if(count==3)
				bhola[a][b]='X';
		return;
}
void Edges(char **gola,char **bhola,int a,int b)
{
	int count=0;
	for(int i=1;i<b;++i)
	{
		
		if(gola[0][i-1]=='X')
			count++;
		if(gola[0][i+1]=='X')
			count++;
		if(gola[1][i]=='X')
			count++;
		if(gola[1][i-1]=='X')
			count++;
		if(gola[1][i+1]=='X')
			count++;
		if(count==1 || count==0 || count ==4)
			bhola[0][i]='-';
		else
			if(count==3)
				bhola[0][i]='X';
		
		count=0;
		if(gola[a][i-1]=='X')
			count++;
		if(gola[a][i+1]=='X')
			count++;
		if(gola[a-1][i]=='X')
			count++;
		if(gola[a-1][i-1]=='X')
			count++;
		if(gola[a-1][i+1]=='X')
			count++;
		if(count==1 || count==0 || count==4)
			bhola[a][i]='-';
		else
			if(count==3)
				bhola[a][i]='X';
	}
	for(int i=1;i<a;++i)
	{
		
		count=0;
		if(gola[i-1][b]=='X')
			count++;
		if(gola[i+1][b]=='X')
			count++;
		if(gola[i-1][b-1]=='X')
			count++;
		if(gola[i][b-1]=='X')
			count++;
		if(gola[i+1][b-1]=='X')
			count++;
		if(count==1 || count==0 || count==4)
			bhola[i][b]='-';
		else
			if(count==3)
				bhola[i][b]='X';
		
		count=0;
		if(gola[i-1][0]=='X')
			count++;
		if(gola[i-1][1]=='X')
			count++;
		if(gola[i][1]=='X')
			count++;
		if(gola[i+1][1]=='X')
			count++;
		if(gola[i+1][0]=='X')
			count++;
		if(count==1 || count==0 || count==4)
			bhola[i][0]='-';
		else
			if(count==3)
				bhola[i][0]='X';
		return;
	}



}
void Center(char **gola,char **bhola,int a,int b)
{
	for(int i=1;i<a;++i)
	{
		for(int y=1;y<b;++y)
		{
			int count=0;
			if(gola[i][y+1]=='X')
				count++;
			if(gola[i][y-1]=='X')
				count++;
			if(gola[i+1][y-1]=='X')
				count++;
			if(gola[i-1][y-1]=='X')
				count++;
			if(gola[i+1][y]=='X')
				count++;
			if(gola[i+1][y+1]=='X')
				count++;
			if(gola[i-1][y+1]=='X')
				count++;
			if(gola[i-1][y]=='X')
				count++;
			if(count==1 || count==0 || count==4)
				bhola[i][y]='-';
			else
				if(count==3)
					bhola[i][y]='X';		
		}
	}
	return;
}
