import java.awt.*;
import java.applet.*;

public class pacman1 extends Applet 
{ 
int xpos=20; int ypos=20; 
int xinc=2; int yinc=2;
int xtop=20; int ytop=20; 
int xbot=350; int ybot=250; 
Color color;


int top = 2;
int left = 2;
int width = 30;
int height = 30;
int startAngle = 290;
int sweepAngle = 320;


public void init()	{ 
color=Color.red;	
}

public void paint(Graphics g){

g.setColor(color);


if(xpos!=xbot && ypos==ybot)
{
xinc =1;
}

else if(xpos<=xbot && ypos!=ybot)
{
yinc=-1;
xpos+=xinc; ypos+=yinc;
}

else if(xpos>=xbot && ypos>=ybot)
{
yinc = -1;
xpos+=xinc; ypos+=yinc;
}
else if(xpos >=xbot && ypos<= ybot)
{
xinc = -1;
xpos+=xinc; ypos+=yinc;
}


g.fillArc(top,xpos, width, height,startAngle,sweepAngle) ;
try{
			Thread.sleep(10);
			}
		catch(Exception e){}
		repaint(10); 


         

}//end paint

}//end class
