#include<iostream>

using namespace std;
class node
{
public:
	int data; 
	node *next;
	node (int a, node *b)
	{
		data =a;
		next =b;
	}

	node(void);
	//~node(void);

};


class linkedlist
{
private:
	node *head;
public:
	void addnode (int );
	void deletenode (int );
	void insertnode (int,int );
	void printnode( );
	linkedlist()
	{
		head=NULL;
	}
	~linkedlist(void);

};

void linkedlist:: deletenode( int index)
{
	node *n;
	node *delnode ;
	node *prevnode;
	node *nextnode;
	n=head;
	for (int i=1; i<index;i++)
	{
		if (i==index-1)
		{
			prevnode=n;}
		n=n->next;
	}
	delnode =n;
	nextnode =delnode->next;
	prevnode->next=nextnode;
	delete delnode;
}


void linkedlist::insertnode(int val,int num)
{
	node *n;
	node* newnode ;
	node *prevnode;
	node *nextnode;
	n=head;
	for (int i=1; i<=val;i++)
	{
		if (i==val-1)
		{
			prevnode=n;}
		n=n->next;
	}
	nextnode =n;
	newnode = new node(num,NULL);
	prevnode->next= newnode;
	newnode->next=nextnode ;
	
	//node createnode;

}

void linkedlist::addnode(int val)
{
	node *n;
	n=head;
	if(head ==NULL)
	{
		head=new node(val,NULL);
	}
	else
	{
		while (n->next!=NULL)
		{
			n=n->next;
		}
		n->next =new node(val,NULL);
	}
}

{
	node *n;

		cout <<"data = "<<n->data<<endl;//display data in n

	}
}
