import java.awt.*;
import java.applet.*;

public class MoveBall extends Applet 
{ 
int xpos=10; int ypos=50; 
int xinc=0; int yinc=1;
int xtop=10; int ytop=10; 
int xbot=350; int ybot=250; 
int a=338; int b=225;
int ainc=10; int bdec=10;
Color color;


public void init()	{ 
color=Color.blue;	
}
public Color genColor() {
return new Color((float)Math.random(),(float)Math.random(),(float)Math.random()); 
}

public void paint(Graphics g){

         	g.setColor(color);
		if(ypos>=ybot)
		{
			ypos=ybot;
			xinc=1;
			a=45; b=270;
			
		}
		
		if(xpos>=xbot)
		{
			xpos=xbot;
			xinc=0;
			yinc=-1;
			a=135; b=290;
			
		}
		if(ypos<=ytop)
		{
			ypos=ytop;
			xinc=-1;
			a=225; b=300;
		}
		if(xpos<=xtop && ypos<=ytop)
		{
			xpos=xtop;
			xinc=0; 
			yinc=1;
			ypos=50;
			a=338; b=225;
		}
		xpos+=xinc;
		ypos+=yinc;
		
		g.fillArc(xpos,ypos,35,35,a,b);
		try{
			Thread.sleep(10);
		   }
		catch(Exception e){}
		repaint(10); 
}//paint ends 

}//class ends
