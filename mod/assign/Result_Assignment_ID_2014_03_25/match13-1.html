#include <stdio.h>

int main()
{
    int m=0;//number
	int l=0;//limit
	int i=0;
	char choice='z';
	printf("**Wel Come To Table**\n**For Multiplication Table Enter 'T'**\n**For Exit Enter 'q'**\n");
	while (choice!='q')
	{
		printf("enter your choice:");
		scanf("%c",&choice);
		switch (choice)
		{
		case'T':
			{
	printf("*****Multiplication Table*****\n");
	printf("Enter number of multiplication Table:");
	scanf("%d",&m);
	printf("Enter Limit of multiplication Table:");
	scanf("%d",&l);
	for (i=1;i<=l;i++)
	{
		printf("%d * %d = %d\n",m,i,m*i);
	}
			}
			}
		}
	getchar();
	return 0;
}
