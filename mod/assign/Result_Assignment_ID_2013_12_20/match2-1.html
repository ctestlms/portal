#include<iostream>
using namespace std;

class Node
{
public:
	int data;
	Node *next;
	Node(int a ,Node *b)
	{
		data=a;
		next=b;
	}
};
class linkedlist
{
public:
	void addnode(int);
	void insertnode(int,int);
	void printnode();
	linkedlist()
	{
	head=NULL;
	}
private:
	Node *head;

};

void linkedlist::addnode(int val)
{
	Node *n;
	n=head;
	if(head==NULL)
	{
		head=new Node(val,NULL);
	}
	else
	{
		while(n->next!=NULL)
		{
			n=n->next;
		}
		n->next=new Node(val, NULL);
	}
}
void linkedlist::printnode()
{
	Node *n;
	int c=1;
	n=head;
	while(n!=NULL)
		{
			cout<<"Data at position "<<c<<"="<<n->data<<endl;
			n=n->next;
	        c++;
	}
}

void linkedlist::insertnode(int pos,int value)
{
	Node *n;
	Node *insertnode;
	Node *prevnode;
	Node *nextnode;
	n=head;
	for(int m=1;m<pos;m++)
	{
		if(m==pos-1)
		{
			prevnode=n;
		}
		n=n->next;
	}
	
	n=new Node(value, n);
	insertnode=n;
	nextnode=insertnode->next;
	prevnode->next=insertnode;
}
