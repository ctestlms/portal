#include <iostream> 
using namespace std;

class Queue
{
    public:
        Queue();
        void EnQueue(int);
        void DeQueue();
        void printQueue();
		~Queue();

	private:
		int HoldData;
        Queue *Front;
		Queue *Last;
        Queue *pointer;
};

Queue::Queue() : Front(NULL) , Last(NULL)
{
}

void Queue::EnQueue(int data)
{
    Queue *temp = new Queue;
    temp->pointer = NULL;
	temp->HoldData = data;
		if(Front == NULL)
		{
			Front = temp;
			Last = temp;
		}
		else
		{
			Last->pointer = temp;
			Last = temp;
		}
	cout<<"Data entered to Queue is: " <<data <<endl;
}

void Queue::DeQueue()
{
    Queue *temp = new Queue;
		if(Front == NULL)
		{
			cout<<"Queue is Empty !";
		}
		else
		{
			temp = Front;
			Front = Front->pointer;
			cout<<"Data removed from Queue is: " <<temp->HoldData;
        
			temp = NULL;
			delete temp;
		}
	cout<<endl;
}

void Queue::printQueue()
{
	cout<<endl;
    Queue *ptr = new Queue;
    ptr = Front;
		if(Front == NULL)
		{
			cout<<"Queue is Empty !";
		}
		else
		{
			cout<<"  Queue is: ";
			while(ptr!=NULL)
			{
				cout<<ptr->HoldData <<" ";
				ptr = ptr->pointer;
			}
		}
	cout<<endl <<endl;
}

Queue::~Queue()
{
	Front = NULL;
	 delete Front;
	Last = NULL;
	 delete Last;
	pointer = NULL;
	 delete pointer;
}
