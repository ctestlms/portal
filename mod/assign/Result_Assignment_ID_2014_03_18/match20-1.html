import java.awt.*;
import java.applet.*;

public class Pman extends Applet 
{ //start of class
int xpos=10; int ypos=15; //initial position
int xinc=3; int yinc=3;
int xbot=370; int ybot=270; 
int xtop=12; int ytop=10; 
Color color;
public void paint(Graphics g){
		g.setColor(color.yellow); //defining colour
		if(ypos<=ybot){ 		//conditions for moving downwards
		ypos
+=yinc;
		g.fillArc(xpos,ypos,30,30,290,330);}
		else if((ypos>=ybot) && (xpos<xbot)){		//condition for moving right
		xpos+=xinc;
		g.fillArc(xpos,ypos,30,30,60,300);}
		else if((xpos>=xbot) && (ypos>=ybot)){		//condition for moving upward
		ypos-=yinc;
		g.fillArc(xpos,ypos,30,30,180,300);
		}
		else if((xpos==xtop) && (ypos==ytop)){		//condition for moving left
		xpos-=xinc;
		g.fillArc(xpos,ypos,30,30,200,300);
		}
		try{
			Thread.sleep(10);		//to slow down pacman's speed
			}
		catch(Exception e){}
		repaint(10); 
}//paint ends 

}//class ends
