import java.awt.*;
import java.applet.*;

public class PacMan1 extends Applet 
{ 
int xpos=10; int ypos=50; 
int xinc=2; int yinc=2;
int xtop=20; int ytop=20; 
int xbot=370; int ybot=270; 
Color color;

public void init()	{ 
color=Color.black;	
}
public Color genColor() {
return new Color((float)Math.random(),(float)Math.random(),(float)Math.random()); 
}

public void paint(Graphics g){

         	if ((ypos<=ybot)&&(xpos==xtop)){
			ypos++;
			g.setColor(Color.black);
			g.fillArc(xpos,ypos,40,40,-45,270);
		}
		else if ((ypos==ybot)&&(xpos!=(xbot-5)))
		{
			xpos+=xinc;
			g.setColor(Color.black);
			g.fillArc(xpos,ypos,40,40,45,270);
		}	
		else if ((ypos!=ytop)&&(xpos==(xbot-5)))
		{
		 	ypos-=yinc;
			g.setColor(Color.black);
			g.fillArc(xpos,ypos,40,40,45,-270); 
		}		 
		else 
		{
		   	xpos-=xinc;	 
		    	g.setColor(Color.black);
			g.fillArc(xpos,ypos,40,40,210,300);
		
}		
			    		   
		
		try{
			Thread.sleep(10);
			}
		catch(Exception e){}
		repaint(10); 
}//paint ends 

}//class ends
