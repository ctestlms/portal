 char** generatecolony(int rows, int cols) {
  char** colony = new char*[rows];

  for (int i = 0; i < rows; i++) {
    colony[i] = new char[cols];
  }

  for (int i = 0; i < rows; i++) {
    for (int j = 0; j < cols; j++) {
      colony[i][j] = rand() % 250;
    }
  }

  return colony;
}

void freeMatrix(char** colony, int rows, int cols) {

	for (int i = 0; i < rows; i++) {
  	 delete colony[i];
  	}
    delete colony;
}
