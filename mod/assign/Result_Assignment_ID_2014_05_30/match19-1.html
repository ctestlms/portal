#include <stdio.h>
#include <stdlib.h>
int main()
{
	int *ptr;

	int i,n1,n2,sum=0;
	float average=0;
	char choice='a';
	printf("enter size of array  =  ");
	scanf("%d",&n1);
	ptr=(int*)malloc(n1*sizeof (int));

	for(i=0;i<n1;i++)
		{
			printf("\nEnter marks of student %d =  ",i+1);
	scanf("%d",&ptr[i]);
	sum+=*(ptr+i);
	
	}
		average=sum/(n1);
	printf("Average = %f",average);
	printf("\npress any key to continue and press q to quit  =  ");
	flushall();
		scanf("%c",&choice);
	while (choice!='q')
	{
		printf("\nenter size of new array   =");
	scanf("%d",&n2);
	ptr=(int*)realloc(ptr,n2);
	for(i;i<n2;i++)
		{
			printf("\nEnter marks of student %d =  ",i+1);
	scanf("%d",&ptr[i]);
	sum+=*(ptr+i);
	}
	average=sum/(n2);
	printf("press any key to continue and press q to quit  =  ");
	flushall();
		scanf("%c",&choice);
	}
	free (ptr);
	return 0;
}
