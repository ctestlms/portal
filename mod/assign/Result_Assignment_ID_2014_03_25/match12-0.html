	while (Option!='q')
	{
		printf("Enter value of Base\t=\t");
		scanf("%d",&Base);
		printf("Enter value of exponent\t=\t");
		scanf("%d",&Exponent);
		for (int counter=1; counter<=Exponent;counter++)
		{
			Result=Result*Base;
		}

	}

}
