#include<stdio.h>


int main()
{
	int arr1[3],arr2[3],arr3[3],arr4[3];
	printf("insert first column:");
	for(int i=0;i<3;i++)
		scanf("%d",&arr1[i]);
	printf("\ninsert second column:");
	for(int i=0;i<3;i++)
		scanf("%d",&arr2[i]);
	printf("\ninsert third column:");
	for(int i=0;i<3;i++)
		scanf("%d",&arr3[i])
	printf("\ninsert vector column:");
	for(int i=0;i<3;i++)
		scanf("%d",&arr4[i]);
	int arr5[3];
	arr5[0]=(arr1[0]*arr4[0])+(arr2[0]*arr4[1])+(arr3[0]*arr4[2]);
	arr5[1]=(arr1[1]*arr4[0])+(arr2[1]*arr4[1])+(arr3[1]*arr4[2]);
	arr5[2]=(arr1[2]*arr4[0])+(arr2[2]*arr4[1])+(arr3[2]*arr4[2]);
	printf("\n resultant matrix is ::\n\t[ %d %d %d]",arr5[0],arr5[1],arr5[2]);
	return 0;
}
