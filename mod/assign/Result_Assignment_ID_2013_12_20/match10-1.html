#include<iostream>
using namespace std;


class node
{
public:
	int data;
	node *next;
	node(int a, node *b)
	{
		data=a;
		next=b;
	}
};
 class list
 {
 private:
	 node *head;
 public:
	 void addnode(int);
	 void delnode(int);
	 void printnode();
	 void insertnode(int,int);
	 list()
	 {
		 head=NULL;

	 }

 };
  
 void list::delnode(int i)
 {
	 node *n;
	 node *delnode;
	 node *prevnode;
	 node *nextnode;
	 n=head;

	 for(int j=1; j<i; j++)
	 {
		 if(j==i-1)
		 {
			 prevnode=n;
		 }
		 n=n->next;
	 }
	 delnode=n;
	 nextnode=delnode->next;
	 prevnode->next=nextnode;
	 delete delnode;

 }

 void list::addnode(int value)
 {
	 node *n;
	 n=head;
	 if(head==NULL)
	 {
		 head=new node(value,NULL);
	 }
	 else
	 {
		 while(n->next!=NULL)
		 {
			 n=n->next;
		 }
		 n->next= new node(value,NULL);
	 }
 }
 void list::insertnode(int val,int pos)

 {
	 node *n;
	 node *newnode;
	 node *prevnode;
	 node *nextnode;
	 n=head;

	 for(int j=1; j<=pos; j++)
	 {
		 if(j==pos)
		 {
			 prevnode=n;
		 }
		 n=n->next;
	 }
	 
	 nextnode=n;
	 newnode=new node(val,NULL);
	 
	 prevnode->next=newnode;
	 newnode->next=nextnode;
	 
 }
