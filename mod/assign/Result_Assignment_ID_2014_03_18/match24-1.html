import java.awt.*;
import java.applet.*;

public class GUIbasics extends Applet 
{ 
int xpos=30; int ypos=30; 
int xinc=2; int yinc=2;
int xtop=20; int ytop=20; 
int xbot=300; int ybot=200; 
int flag = 0;
Color color;


public void init()	{ 
color=Color.yellow;	
}
public Color genColor() {
return new Color((float)Math.random(),(float)Math.random(),(float)Math.random()); 
}

public void paint(Graphics g){

         
		if (ypos <= ybot && xpos >= xtop){
			xpos += 0;
			yinc = 2; xinc =2;
			ypos += yinc;
			g.fillArc(xpos, ypos, 50, 50, 325, 275);
		}
		else if (xpos >= xbot && ypos >= ytop){
			xpos += 0;
			yinc = -2; xinc =2;
			ypos += yinc;
			g.fillArc(xpos, ypos, 50, 50, 25, 275);
		}
		else if(ypos >= ybot && xpos >= xtop){
			yinc = 2; xinc =2;
			xpos += xinc;
			ypos += 0;
			g.fillArc(xpos, ypos, 50, 50, 50, 275);
		}
		else if (xpos <= xbot && ypos <= ytop){
			ypos += 0;
			xinc = -2; yinc = 2;
			xpos += xinc;
			g.fillArc(xpos, ypos, 50, 50, 50, 275);
		
}

		g.setColor(color);
//		g.fillOval(xpos,ypos,25,25);


		try{
			Thread.sleep(10);
			}
		catch(Exception e){}
		repaint(10); 
}//paint ends 

}//class ends
