/* Used for visualizing applet on screen */

import java.awt.*;
import java.applet.*;

public class Pacman extends Applet{ /* 'Applet' is a class which we are extending*/

	int xpos=10; int ypos=10;
	int txpos=400; int typos=300;
	Color color = Color.red;

	public void paint( Graphics g ){ /* g is an object of the Graphics class*/
		g.setColor(color);

			/* the following checks allow the pacman to move in a rectangle */
		if( ypos <= typos ){
		ypos = ypos + 5;
		g.fillArc(xpos, ypos, 20, 20, 300, 315);
		}

		 if( ypos >= typos && xpos <= txpos){
		g.fillArc(xpos, ypos, 20, 20, 30, 300);
		xpos = xpos + 5;
		}
		 if( ypos >= 10 && xpos == txpos){
			ypos = ypos - 5;
			g.fillArc(txpos, typos, 20, 20, 70, 310);
		}
	
		try{ 
		Thread.sleep(10);
		}
		catch(Exception e){}
		repaint(10); 
}
}
