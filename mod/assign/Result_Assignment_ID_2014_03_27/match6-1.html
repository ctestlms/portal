int main()
{
 float Farenheit;
 float Celcius;
 float Kilogram;
 float Pound;
 float Dollars;
 float Rupees;
char choice = 'i';
while (choice!='e')
{
	printf("Enter a Char (e for exit; T for Temprature; W for Weight; C for Currecny) = ");
	scanf(" %c",&choice);


	switch(choice)
	{
	case 'T':
		printf("Enter temperature in Celcius=");
		scanf("%f",&Celcius);

		Farenheit = Celcius*9/5 + 32;
		printf("Temperature in Farenheit=%f\n",Farenheit);
             
		break;
	case 'W':
		printf("Enter Weight in Pounds=");
		scanf("%f",&Pound);
		Kilogram=0.453592*Pound;
		printf("Weight in Kilograms=%f\n",Kilogram);
		
		break;
       case 'C':
       	printf("Enter Currency in PKR=");
		scanf("%f",&Rupees);
		Dollars = 99/Rupees;
		printf("Currency in Dollars=%f\n",Dollars);
		break;


	default:
		printf("Wrong input =%c\n",choice);

	}
	
}
  getche();
  return 0;
}
