#include <stdio.h>
#include <stdlib.h>
#include <conio.h>

int main()
{
	int *ptr,n1=0,i,n2=0,class_average,total=0,b=1,n3=0;
	char choice = 'i';
	char boolean ='Y';
	while (choice!='q')
	{
		printf("Press any key to continue OR press q to quit and display results...\n");
		scanf(" %c",&choice);
		switch(choice)
		{
		case 'q':
			if(n3==0)
			{
			class_average=total/(n1+n2);
			printf("Class average = %d",class_average);
			}
			else
			{
			class_average=total/(n1+n3);
			printf("Class average = %d",class_average);
			}

			break;
		default:
			printf("Enter the number of records you would like to enter...\n");
			scanf(" %d",&n1);
			ptr=(int*)malloc(n1*sizeof(int));
			for(i=0;i<n1;i++)
			{
				printf("Enter marks for student %d\n",i+1);
				scanf("%d",ptr+i);
				total=total+*(ptr+i);
			}
			printf("Would you like to enter more records ?\nPress Y for Yes or N for no...\n");
			scanf(" %c",&boolean);
			while(boolean=='Y')
			{
				printf("How many more records would you like to enter?...\n");
				scanf("%d",&n2);
				n3=n3+n2;
				ptr=(int*)realloc(ptr,(n1+n3)*sizeof(int));
				for(i;i<(n1+n3);i++)
				{
					printf("Enter marks for student %d\n",i+b);
					scanf("%d",ptr+i);
					total=total+*(ptr+i);
				}
				printf("Would you like to enter more records ?\nPress Y for Yes or N for no...\n");
				scanf(" %c",&boolean);
			}
			break;
		}}

	getch();
	return 0;
}
