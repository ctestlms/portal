	float z=0;      // currency in USD

	while (choice!='E')  // if the value of choice is entered E then programe will be terminate

		switch(choice)

		case 'A':

			printf("Enter temparature in celsius\n");
			scanf("%f",&a);
			b=a*9/5 + 32;
			printf("temparature in Farenhite is%f\n",b);

				printf("Enter temparature in Farenhite\n");
