import java.awt.*;
import java.applet.*;

public class MoveBall extends Applet 
{ 
int xpos=10; int ypos=20; 
int xinc=0; int yinc=3;
int xtop=10; int ytop=30; 
int xbot=370; int ybot=270; 
Color color;


public void init()	{ 
color=Color.red;	
}
int i=280;
int j=335;
public void paint(Graphics g){

         	
		if (xpos==xbot){   yinc=-3; xinc=0; }
		else if(xpos<xtop) { xinc=-3; yinc=0;  }
		
		else if (ypos>ybot) {  xinc=3;  yinc=0;  }
		else if(ypos==ytop) {  yinc=-3;   xinc=0;
}
		
			//else if(xpos==xtop)  { xinc=-3; yinc=0; }
		xpos+=xinc; ypos+=yinc;
		g.fillArc(xpos, ypos, 30,30 , i,j);
		
		try{
			Thread.sleep(40);
			}
		catch(Exception e){}
		repaint(10); 
}//paint ends 

}//class ends
