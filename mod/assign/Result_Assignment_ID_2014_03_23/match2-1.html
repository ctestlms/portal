#include<stdio.h>


int main()
{
float cel;
float fah;
float kilo;
float lbs;
float PKR;
float USD;
char choice = 'i';
while (choice!='e')
{
printf("Enter a Char (T for Temprature; W for Weight; C for money\n");
scanf(" %c",&choice);
	switch(choice)
	{
	case'T':
    printf("Enter Temperature in celsius\n");
    scanf("%f",&cel);
	fah= 32+(cel * 9/5);
	printf ("temperature in fah is %f\n",fah);
	break;

	case'W':
    printf("Enter weight in lbs\n");
    scanf("%f",&lbs);
	kilo= 0.4536*lbs;
	printf ("weight in kilo is %f\n",kilo);
	break;

	case'C':
	printf("Enter money in USD\n");
    scanf("%f",&USD);
	PKR = 99*USD ;
	printf ("money in PKR is %f\n",PKR);
	break;

	default:
	printf("Wrong input =%c\n",choice);
	}
	}

}
