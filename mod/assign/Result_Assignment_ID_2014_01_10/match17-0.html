#include<iostream>
using namespace std;
class item
{
public:
	int Data; 
	item *Next;
	item (int value, item *pntr)
	{
		Data =value;
		Next =pntr;
	}

	item(void);
};


class link_list
{
private:
	item *head;
public:
	void additem (int );
	void deleteitem (int );
	void insertitem (int,int );
	void printitem( );
	link_list()
	{
		head=NULL;
	}
	~link_list(void);
};

void link_list::insertitem(int val,int num)
{
	item *n;
	item* newitem ;
	item *previtem;
	item *Nextitem;
	n=head;
	for (int i=1; i<=val;i++)
	{
		if (i==val-1)
		{
			previtem=n;
		}
		n=n->Next;
	}
	Nextitem =n;
	newitem = new item(num,NULL);
	previtem->Next= newitem;
	newitem->Next=Nextitem ;
}

void link_list::additem(int val)
{
	item *n;
	n=head;
	if(head ==NULL)
	{
		head=new item(val,NULL);
	}
	else
	{
		while (n->Next!=NULL)
		{
			n=n->Next;
		}
		n->Next =new item(val,NULL);
	}
}
void link_list:: printitem()
{
	item *n;
	n=head;
	int cont=1;
	while (n!=NULL)
	{
		cout <<"Data of Item "<<cont<<" is "<<n->Data<<endl;
		n=n->Next;
		cont++;
	}
}
link_list::~link_list(void)
{
}
