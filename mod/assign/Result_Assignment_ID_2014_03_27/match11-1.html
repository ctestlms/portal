#include<stdio.h>
#include<conio.h>

int main()
{
    float fahr=0;
	float celcius=0;
	float kelvin=0;
	float lb=0;
	float kg=0;
	float PKR=0;
	float USD=0;
	float sq_mile=0;
	float sq_km=0;
	float cubic_inch=0;
	float cubic_cm=0;
	char choice=0;
	while (choice !='e')
	{
		printf("Welcome to my converter application!\nPress a,b,h,i for temperature conversions\nPress c,d for mass calculations\nPress f,g for currency conversions\nPress j,k for area conversions\nPress l,m for volume conversions\n");
		scanf("%c", &choice);
		switch (choice)
		{
		case 'a':
			printf("Temperature in fahr?\n");
			scanf("%f",&fahr);
			celcius=(fahr*5/9)-32;
			printf("The temperature in celcius is %3.2f\n",celcius);
			break;
		case 'b':
			printf("Temperature in Celcius?\n");
			scanf("%f",&celcius);
			fahr=(celcius*9/5)+32;
			printf("The temperature in fahrenheit is %3.2f\n",fahr);
			break;
		case 'c':
			printf("Weight in pounds?\n");
			scanf("%f",&lb);
			kg=lb*0.453592;
			printf("Weight in kg is 3.2%f\n",kg);
			break;
		case 'd':
			printf("Weight in kg?\n");
			scanf("%f",&kg);
			lb=kg/0.453592;
			printf("Weight in lb is 3.2%f\n",lb);
			break;
		case 'f':
			printf("Enter currency in USD:\n");
			scanf("%f",&USD);
			PKR=USD*99;
			printf("The value in PKR is 3.2%f\n",PKR);
			break;
		case 'g':
			printf("Enter currency in PKR:\n");
			scanf("%f",&PKR);
			USD=PKR/99;
			printf("The value in USD is 3.2%f\n",USD);
			break;
		case 'h':
			printf("Enter Temperature in kelvin:\n");
			scanf("%f",&kelvin);
			celcius=kelvin-273;
			printf("The temperature in celcius is 3.2%f\n",celcius);
			break;
		case 'i':
			printf("Enter Temperature in celcius:\n");
			scanf("%f",&celcius);
			kelvin=celcius+273;
			printf("The temperature in kelvin is 3.2%f\n",kelvin);
			break;
		case 'j':
			printf("Area in square miles?\n");
			scanf("%f",&sq_mile);
			sq_km=sq_mile*2.26;
			printf("Area in square km is 3.2%f\n",sq_km);
			break;
		case 'k':
			printf("Area in square km?\n");
			scanf("%f",&sq_km);
			sq_mile=sq_km/2.26;
			printf("Area in square mile is 3.2%f\n",sq_mile);
			break;
		case 'l':
			printf("Volume in Cubic inches?\n");
			scanf("%f",&cubic_inch);
			cubic_cm=cubic_inch*16.39;
			printf("Volume in Cubic cm is 3.2%f\n",cubic_cm);
			break;
		case 'm':
			printf("Volume in Cubic cm?\n");
			scanf("%f",&cubic_cm);
			cubic_inch=cubic_cm/16.39;
			printf("Volume in Cubic inch is 3.2%f\n",cubic_inch);
			break;
		default:
			printf("Invalid input\nTry again\n");
	}	
	}
	getchar();
		return 0;
}
