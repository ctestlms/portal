 char** generategrid(int rows, int cols) {
  char** mygrid = new char*[rows];

  for (int i = 0; i < rows; i++) {
    mygrid[i] = new char[cols];
  }

  for (int i = 0; i < rows; i++) {
    for (int j = 0; j < cols; j++) {
      mygrid[i][j] = rand() % 250;
    }
  }

  return mygrid;
}

void freeMatrix(char** mygrid, int rows, int cols) {

    for (int i = 0; i < rows; i++) {
       delete mygrid[i];
      }
    delete mygrid;
}
