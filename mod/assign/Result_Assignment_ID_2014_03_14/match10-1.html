void Corner(char **s1,char **s2,int a,int b)
{
	//first corner
	int count=0;
	if(s1[0][1]=='X')
		++count;
	if(s1[1][1]=='X')
		++count;
	if(s1[1][0]=='X')
		++count;
	if(count==1 || count==0)
	{
		s2[0][0]='-';
	}
	else
		if(count==3){
			s2[0][0]='X';
		}//second corner
		count=0;
		if(s1[a][1]=='X')
			++count;
		if(s1[a-1][1]=='X')
			++count;
		if(s1[a-1][0]=='X')
			++count;

		if(count==1 || count==0)
			s2[a][0]='-';
		else
			if(count==3)
				s2[a][0]='X';
		//third corner
		count=0;
		if(s1[0][b-1]=='X')
			++count;
		if(s1[1][b-1]=='X')
			++count;
		if(s1[1][b]=='X')
			++count;
		if(count==1 || count==0)
			s2[0][b]='-';
		else
			if(count==3)
				s2[0][b]='X';
		//fourth corner

		count=0;
		if(s1[a][b-1]=='X')
			++count;
		if(s1[a-1][b-1]=='X')
			++count;
		if(s1[a-1][b]=='X')
			++count;
		if(count==1 || count==0)
			s2[a][b]='-';
		else
			if(count==3)
				s2[a][b]='X';
		return;
}
void Edge(char **s1,char **s2,int a,int b)
{
	int count=0;

	for(int i=1;i<b;++i)
	{
		//upper Edge
		if(s1[0][i-1]=='X')
			count++;
		if(s1[0][i+1]=='X')
			count++;
		if(s1[1][i]=='X')
			count++;
		if(s1[1][i-1]=='X')
			count++;
		if(s1[1][i+1]=='X')
			count++;
		if(count==1 || count==0 || count >=4)
			s2[0][i]='-';
		else
			if(count==3)
				s2[0][i]='X';
		//lower Edge
		count=0;

		if(s1[a][i-1]=='X')
			count++;
		if(s1[a][i+1]=='X')
			count++;
		if(s1[a-1][i]=='X')
			count++;
		if(s1[a-1][i-1]=='X')
			count++;
		if(s1[a-1][i+1]=='X')
			count++;
		if(count==1 || count==0 || count>=4)
			s2[a][i]='-';
		else
			if(count==3)
				s2[a][i]='X';
	}
	for(int i=1;i<a;++i)
	{
		//right Edge
		count=0;

		if(s1[i-1][b]=='X')
			count++;
		if(s1[i+1][b]=='X')
			count++;
		if(s1[i-1][b-1]=='X')
			count++;
		if(s1[i][b-1]=='X')
			count++;
		if(s1[i+1][b-1]=='X')
			count++;
		if(count==1 || count==0 || count>=4)
			s2[i][b]='-';
		else
			if(count==3)
				s2[i][b]='X';
		//left Edge
		count=0;

		if(s1[i-1][0]=='X')
			count++;
		if(s1[i-1][1]=='X')
			count++;
		if(s1[i][1]=='X')
			count++;
		if(s1[i+1][1]=='X')
			count++;
		if(s1[i+1][0]=='X')
			count++;
		if(count==1 || count==0 || count>=4)
			s2[i][0]='-';
		else
			if(count==3)
				s2[i][0]='X';
		return;
	}



}

void Centre(char **s1,char **s2,int a,int b)
{
	for(int i=1;i<a;++i)
	{
		for(int y=1;y<b;++y)
		{
			int count=0;

			if(s1[i][y+1]=='X')
				count++;
			if(s1[i][y-1]=='X')
				count++;
			if(s1[i+1][y-1]=='X')
				count++;
			if(s1[i-1][y-1]=='X')
				count++;
			if(s1[i+1][y]=='X')
				count++;
			if(s1[i+1][y+1]=='X')
				count++;
			if(s1[i-1][y+1]=='X')
				count++;
			if(s1[i-1][y]=='X')
				count++;
			if(count==1 || count==0 || count>=4)
				s2[i][y]='-';
			else
				if(count==3)
					s2[i][y]='X';
		}
	}
	return;
}
