#include<iostream>
using namespace std;
class node
{
public:
	int data;
	node *next;
	node() //default constructor
	{
		data=0;
		next=NULL;
	}
};
class queue
{
private:
	node* head;
public:
	queue() //constructor
	{
		head=NULL;
	}
	void isempty()
	{
		if(head==NULL)
			cout<<"queue is empty"<<endl;
	}

	void ins_at_end(int y) //insert a new node at the last
	{
		node*Nnode;
		Nnode=new node;
		Nnode->data=y;
		if(head==NULL)
			head=Nnode;
		else
		{
			node *temp;
			temp=head;
			while(temp->next!=NULL)
			{
				temp=temp->next;
			}
			temp->next=Nnode;
		}
	}

	void del_end() //delete the last node
	{
		node* temp;
		node* del;
		temp=head;
		if(head==NULL)
		{
				cout<<"the queue is already empty()"<<endl;
				delete head;
		}
		else if(head->next==NULL)
		{
			delete head;
		}
		else
		{
		while(temp->next->next!=NULL)
				temp=temp->next;
		del=temp->next;
		delete del;
		temp->next=NULL;
		
		}
	}
	void traverse() // to print the data at all the nodes
	{
		int n=1;
		node* temp;
		temp=head;
		while(temp!=NULL)
		{
			cout<<"data at node "<<n<<" is: ";
			cout<<temp->data<<endl;
			temp=temp->next;
			n++;
		}
	}
	~queue() //destructor
	{
		delete head;
	}
};
int main()
{
	queue q1;
	q1.ins_at_end(1);
	q1.ins_at_end(2);
	q1.ins_at_end(3);
	q1.ins_at_end(4);
	q1.ins_at_end(5);
	q1.traverse();
	cout<<endl<<endl;

	cout<<"after deleting the last node"<<endl;
	q1.del_end();
	q1.traverse();
	cout<<endl<<endl;

	cout<<"adding a new node"<<endl;
	q1.ins_at_end(6);
	q1.traverse();
	cout<<endl<<endl;
	
	queue q2;
	q2.isempty();
}
