#include <stdio.h>
#include <stdlib.h>
char** memalloc(int rows, int cols);
int** memalloc1(int rows, int cols);
void main()
{
	FILE *fptr;
	int rows, columns;
	printf("Welcome to a simulation of the lifecycle of a bacteria colony. \nCells (X) live and die by the following rules: \n- A cell with 1 or fewer neighbors dies. \n- Locations with 2 neighbors remain stable. \n- Locations with 3 neighbors will create life. \n- A cell with 4 or more neighbors dies.\n");
	printf("Grid input file name?");
	char fname[50];
	gets(fname);
	fptr=fopen(fname, "r");
	fscanf(fptr, "%d\n%d\n", &rows, &columns);
	char **matrix = memalloc(rows, columns);
	int **mat1= memalloc1(rows,columns); //For memory allocation of a temporary array.
	rewind(fptr);
	fscanf(fptr, "%d\n%d\n", &rows, &columns);
	char temp;
	for (int i = 0; i < rows; i++) //Storing the file in array.
	{
		for (int j = 0; j < columns; j++)
		{
			fscanf(fptr, "%c", &matrix[i][j]);
		} 
			fscanf(fptr, "%c", &temp);
	}
	for (int i = 0; i < rows; i++)
	{
		
		for (int j = 0; j < columns; j++)
		{
			printf("%c", matrix[i][j]);
		}	printf("\n");
	}
	while (1)
	{
	printf("t)ick, q)uit? "); 
	char opt;
	fflush(stdin);
	scanf("%c", &opt);
	if (opt=='t')
	{
		int check=0; // Checking neighbour cells.
		for (int i=0;i<rows;i++)
		{
			for (int j=0;j<columns;j++)
			{
				if (i>0 && j>0)
				{
					if (matrix[i-1][j-1]=='X')
						check++;
				}
				if (i>0)
				{
					if (matrix[i-1][j]=='X')
						check++;
				}
				if (j>0)
				{
					if (matrix[i][j-1]=='X')
						check++;
				}
				if (i<rows-1)
				{
					if (matrix[i+1][j]=='X')
						check++;
				}
				if (j>0 && i<rows-1)
				{
					if (matrix[i+1][j-1]=='X')
						check++;
				}
				if (j<columns-1)
				{
					if (matrix[i][j+1]=='X')
						check++;
				}
				if (i>0)
				{
					if (matrix[i-1][j+1]=='X')
						check++;
				}
				if (i<rows-1 && j<columns-1)
				{
					if (matrix[i+1][j+1]=='X')
						check++;
				}
					mat1[i][j]=check;
					check=0;
			}
		}	
	}
	for (int i=0;i<rows;i++) // Printing the next generation.
	{
		for (int j=0;j<columns;j++)
		{	if (mat1[i][j]<2)
			{
				matrix[i][j]='-';
				printf("%c", matrix[i][j]);	
			}
			if (mat1[i][j]==2)
			{
				printf("%c", matrix[i][j]);
			}
			if (mat1[i][j]>=4)
			{
				matrix[i][j]='-';
				printf("%c", matrix[i][j]);
			}
			if (mat1[i][j]==3)
			{
				matrix[i][j]='X';
				printf("%c", matrix[i][j]);	
			}
		}
		printf("\n");
	}
	if (opt == 'q')
		return;
	
	if (opt != 't' && opt != 'q') //For wrong user option.
	{
		printf("Wrong input, enter the choice again.\n");
		continue;
	}
	}
}
char** memalloc(int rows, int cols)
{
	char** matrix= new char*[rows];
	for (int i = 0; i < rows; i++)
	{
		matrix[i] = new char[cols];
	}
	return matrix;
}
int** memalloc1(int rows, int cols)
{
	int** mat1= new int*[rows];
	for (int i = 0; i < rows; i++)
	{
		mat1[i] = new int[cols];
	}
	return mat1;
}
