#include <iostream>
using namespace std;
struct node
{
   int num;
   node *link;
};
class Tqueue
{
 private:
    node *startPt,*endPt;

 public:
    Tqueue()
    {
        startPt=NULL;
        endPt=NULL;
    }
    void add(int n)
    {
        node *temp;
        temp=new node;
       if(temp==NULL)
           cout<<"\nqUEUE FULL";
        temp->num=n;
        temp->link=NULL;
        if(startPt==NULL)
        {
            endPt=startPt=temp;
            return;
        }
           endPt->link=temp;
            endPt=endPt->link;
    }
    int del()
    {
        if(startPt==NULL)
        {
            cout<<"\nqUEUE EmPTY";
            return NULL;
        }
        node *temp;
        int n;
        n=startPt->num;
        temp=startPt;
        startPt=startPt->link;
       delete temp;
        return n;
    }
    ~Tqueue()
    {
       if(startPt==NULL)
           return;
        node *temp;
        while(startPt!=NULL)
        {
            temp=startPt;
            startPt=startPt->link;
            delete temp;
        }
	}
};

int qain()
{
    Tqueue q;
    q.add(1);
    q.add(2);
    q.add(3);
    cout<<"Item Deleted = "<<q.del();
    cout<<"\nItem Deleted = "<<q.del();
    cout<<"\nItem Deleted = "<<q.del()<<"\n";
    return 0;
}
