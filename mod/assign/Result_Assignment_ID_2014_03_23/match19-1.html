#include<stdio.h>
#include<conio.h>
int main()
{
	float temp=0;

	int a=0;
	float temp2=0;
	float weight=0;
	float currency=0;
	float area=0;
	float volume=0;
	char choice = 'X';


while (choice!='T')
{
	printf("\n\t **CONVERTER APPLICATION**\n");
	printf("A) Temperature (Fahrenheit - Celsius)\n");
	printf("B) Temperature (Kelvin - Celsius)\n");
	printf("C) Weight (Pound - Kilo)\n");
	printf("D) Currency (PKR - USD)\n");
	printf("E) Area (Sqr Miles - Sqr Kilometers)\n");
	printf("F) Volume (Cubic Inches - Cubic Centimeters)\n");
	printf("T) Terminate the application\n");
	printf("\nChoose one of the above options: ");
	scanf(" %c",&choice);

	switch(choice)
	{
	case'A':
		printf("Enter the temperature in Fahrenheit: ");
		scanf("%f",&temp);
		temp=(temp-32)/1.8;
		printf("Temperature in Celsius is: %1.2f\n", temp);
		break;
	case'B':
		printf("Enter the temperature in Kelvin: ");
		scanf("%f",&temp2);
		temp2=temp2-273.15;
		printf("Temperature in Celsius is: %1.2f\n", temp2);
		break;
	case'C':
		printf("Enter the weight in Pounds: ");
		scanf("%f",&weight);
		weight=weight*0.453592;
		printf("Weight in Kilograms is: %1.3f\n", weight);
		break;
	case'D':
		printf("Enter the Currency in PKR: ");
		scanf("%f",&currency);
		currency=currency/99;
		printf("Currency in USD is: %1.2f\n", currency);
		break;
	case'E':
		printf("Enter the Area in Square Miles: ");
		scanf("%f",&area);
		area=area*2.59;
		printf("Area in Square Kilometers: %1.2f\n", area);
		break;
	case'F':
		printf("Enter the Volume in Cubic Inches: ");
		scanf("%f",&volume);
		volume=volume*16.387064;
		printf("Area in Cubic Centimeters: %1.2f\n", volume);
		break;
	case'T':
		printf("This application will terminate");
		break;
	default:
		printf("Wrong Input, Choose from the following\nTip: All cases are in uppercase letters\n\n");
	}	
	
}
getche();
return 0;
}
