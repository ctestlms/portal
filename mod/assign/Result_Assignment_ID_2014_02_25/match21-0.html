import java.util.*;
public class Task
{
 public static void main(String args[])
 {
   String[][] data = new String[10][2];
   float[] two = new float[10];
   Scanner keyboard = new Scanner(System.in);

   float sum = 0;
   float least = 0;
   float most = 0;
   float temp = 0;
   String search = null;
   int flag = 0;
   int position=0;
   float gpa=0;

   outer:
   {
   for(int i=0;i<data.length;i++)
   {
    for(int j=0;j<data[i].length;j++)
    {
     System.out.print("Kindly Enter Student No. " + (i+1) + "'s Name: ");
     data[i][j] = keyboard.nextLine();

     for(int k=0;k<i;k++)
     {
      if(data[i][0].equalsIgnoreCase(data[k][0]))
        {
         System.out.println("Dual Entry. Program Aborting.");
         break outer;
        }
     }

     System.out.print("Kindly Enter Student No. " + (i+1) + "'s CGPA: ");
     data[i][++j] = keyboard.nextLine();
     
     float x = Float.parseFloat(data[i][j]);
     
     if(x<0 || x>4)
     {
      System.out.println("Invalid Entry. Program Aborting.");
      break outer;
     }

     sum = sum + x;

    }
   }

   float average = sum/10;

   System.out.println("Name\t\tCGPA");   

   for(int i=0;i<data.length;i++)
   {
    for(int j=0;j<data[i].length;j++)
    {
     System.out.print(data[i][j] + "\t\t");
     System.out.print(data[i][++j]);
    }
    System.out.println("");
   }

    least = Float.parseFloat(data[0][1]);
    most = Float.parseFloat(data[0][1]);
    for(int i=0;i<10;i++)
    {
     float x = Float.parseFloat(data[i][1]);
     if(x<least)
      least=x;
     if(x>most)
      most=x;
    } 

      

   System.out.println("Average of the class is: " + average);
   System.out.println("Minimum CGPA in the class is: " + least);
   System.out.println("Maximum CGPA in the class is: " + most);

   System.out.println("Students who have scored less than the class average: ");
   for(int i=0;i<10;i++)
   {
    temp = Float.parseFloat(data[i][1]);
    if(temp<average)
      System.out.println(data[i][0]);
   }

   
   for(int i=1; i<data.length; i++)
   {
   float x = Float.parseFloat(data[i][1]);
   two[i] = x;
   }

   Arrays.sort(two);


   System.out.println("Enter name of student whose data is required: ");
   search = keyboard.nextLine();

   for(int i=0; i<data.length; i++)
    {
     if(data[i][0].equals(search))
     {
     flag = 1;
     for(int a=0;a<two.length;a++)
     {
      float x = Float.parseFloat(data[i][1]);
      if(x == two[a])
        {
          position=11-a;
          gpa=two[a];
        }
     }
     }         
    }

    if(flag==1)
     System.out.println("Record Found. GPA is " + gpa + ". Class position is " + position);
    else
     System.out.println("Match Not Found.");
  } 
 }
}
