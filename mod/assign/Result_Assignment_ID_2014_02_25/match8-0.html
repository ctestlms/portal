import java.util.*;
import java.lang.*;

public class MyCodee {

		public static void main(String[] args) {
	
		Scanner in=new Scanner(System.in);
		

		float gpa,temp,min=4,max=0,average=0,sum=0;
		String check;
		String[][] data=new String[5][2];
		
		
			for(int i=0;i<data.length;i++)
			{
				for(int j=0;j<data[i].length;j++)
				{
					System.out.println("Enter the name of Student No. "+(i+1)+": ");
					data[i][j]=in.nextLine();
				
					check=data[i][j];
				
					for(int k=0;k<data.length;k++)
					{
						for(int l=0;l<data[k].length;l++)
						{
							if((i==k) && (j==l))
							{
								continue;
							}
							else
							{
								if(check.equalsIgnoreCase(data[k][l]))
								{
									do
									{
										System.out.println("ERROR!");
										System.out.println("DUPLICATE ENTRY");
										System.out.println("TRY AGAIN");
										data[i][j]=in.nextLine();
										check=data[i][j];
									}
									while(check.equalsIgnoreCase(data[k][l]));
								}	
							}
						}
					}
					
					System.out.println("Enter the CGPA of Student No. "+(i+1)+": ");
					data[i][++j]=in.nextLine();
				
					gpa=Float.parseFloat(data[i][j]);
				
					if(!(gpa<0.00|gpa>4.00))
					{
						temp=gpa;
						if(temp<min)
						{
							min=temp;
						}
						if(temp>max)
						{
							max=temp;
						}
						sum+=temp;
					}

					if(gpa<0.00|gpa>4.00)
					{
						do
						{
							System.out.println("Invallid Entry!");
							System.out.println("TRY AGAIN");
							data[i][j]=in.nextLine();
							gpa=Float.parseFloat(data[i][j]);

							if(!(gpa<0.00|gpa>4.00))
								{
									temp=gpa;
									if(temp<min)
									{
										min=temp;
									}
									if(temp>max)
									{
										max=temp;
									}
									sum+=temp;
								}
							
						}
						while (gpa<0.00|gpa>4.00);
					}
				}
			}

			System.out.println("");
			System.out.println("Name\t\tCGPA");
			for(int i=0;i<data.length;i++)
			{
				for(int j=0;j<data[i].length;j++)
				{
					System.out.println(""+data[i][j]+"\t"+data[i][++j]);
				}
			}
			average=sum/5;
			System.out.println("");
			System.out.println("Minimum CGPA = "+min);
			System.out.println("Maximum CGPA = "+max);
			System.out.println("Average CGPA = "+(average));
		
			System.out.println("");
			System.out.println("Following Students got less than average marks.");
			for(int x=0;x<data.length;x++)
			{
				for(int y=0;y<data[x].length;y++)
				{
					gpa=Float.parseFloat(data[x][++y]);
				
					if(gpa<average)
					{
						System.out.println(""+data[x][y-1]);
					}

				}
			}
		

				System.out.println("");			
				System.out.println("Enter the name of Student you want to search: ");
				check=in.nextLine();
				for(int i=0;i<data.length;i++)
				{
					for(int j=0;j<data[i].length;j++)
					{
						
				
						
									if(check.equalsIgnoreCase(data[i][j]))
									{
											System.out.println("");
											System.out.println("Match Found!");
											System.out.println("Name\t\tCGPA");											
											System.out.println(""+data[i][j]+"\t"+data[i][++j]);										
										
									}	
								
							
						
					}
				}

				
	}

}
